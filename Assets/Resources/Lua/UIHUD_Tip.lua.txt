local tbWnd = Ui:CreateWindow("UIHUD_Tip")
local tbControls = {
  Template = "imgBG/List/Viewport/Content/Template"
}
tbWnd.tbControls = tbControls

function tbWnd:NotifyEvents()
  return {
    {
      EventNotify.emNOTIFY_UI_UIHUDTIP_UPDATE,
      self.Update
    }
  }
end

function tbWnd:OnCreate()
  self.tbList = Ui:NewScrollView(self, tbControls.Template)
end

function tbWnd:OnOpen(tbInfo, nMoney, FnCallBack, ...)
  self.tbInfo = tbInfo
  self.FnCallBack = FnCallBack
  self.tbParam = table.pack(...)
  self:Update(nMoney)
end

function tbWnd:Update(nMoney)
  self.nHave = nMoney
  self.tbList:Clear()
  for nIndex, tbData in ipairs(self.tbInfo) do
    self.tbList:SetElement(nIndex, function(tbElement)
      self:_SetElement(tbElement, tbData)
    end)
  end
end

local tbElementPath = {
  UIItemGrid = "UIItemGrid",
  imgMoney = "Cost/imgMoney",
  txtCost = "Cost/txtCost",
  btnOk = "btnOk"
}

function tbWnd:_SetElement(tbElement, tbData)
  local pPanel = tbElement.pPanel
  local tbGrid = Ui:NewExistItemGrid(tbElement, tbElementPath.UIItemGrid)
  tbGrid:SetItemByTemplate(tbData.dwTemplateID, nil, nil, nil, {bShowTip = true})
  tbGrid:SetClickEnable(true)
  pPanel:Sprite_SetSprite(tbElementPath.imgMoney, tbData.szIcon)
  pPanel:Label_SetText(tbElementPath.txtCost, string.format("%d/%d", self.nHave or 0, tbData.nPrice))
  pPanel:Button_SetText(tbElementPath.btnOk, i18n.EasyGet("\228\189\191\231\148\168"), "Text")
  Ui:Button_BindEvent(pPanel, tbElementPath.btnOk, self._OnClickButton, self, tbData.dwTemplateID)
end

function tbWnd:_OnClickButton(dwTemplateID)
  Lib:CallBack({
    self.FnCallBack,
    table.unpack(self.tbParam),
    dwTemplateID
  })
end

function tbWnd:CloseByDefault()
end
