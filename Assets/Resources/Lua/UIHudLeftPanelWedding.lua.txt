local tbWnd = Ui:CreateClass("UIHudLeftPanelWedding", "UIPanel")
tbWnd.nContinueTime = 0
tbWnd.tbControls = {
  txtTimer = "MyRank/txtTimer",
  txtTips = "MyRank/txtTips",
  txtBottom = "MyRank/txtBottom",
  txtContentTime1 = "UIDefaultFubenInfo/InfoScroll/Viewport/Content/txtContent1/txtContentTime1",
  txtContent2 = "UIDefaultFubenInfo/InfoScroll/Viewport/Content/txtContent2",
  txtContentTime = "Image/txtContentTime",
  txtContentInfo = "Image/txtContentTime/txtContentInfo",
  txtContentInfo1 = "Image/txtContentInfo1"
}
tbWnd.tbOnClick = {}

function tbWnd:OnCreate()
end

function tbWnd:OnOpen()
end

function tbWnd:OnClose()
end

function tbWnd:OnActivate()
  if self.nTimerId then
    Timer:Close(self.nTimerId)
    self.nTimerId = nil
  end
  self:SynTimeInfo()
  self.nTimerId = Timer:Register(Env.GAME_FPS, self.ShowTime, self)
end

function tbWnd:OnDeactivate()
  if self.nTimerId then
    Timer:Close(self.nTimerId)
    self.nTimerId = nil
  end
end

function tbWnd:UpdateData(...)
  local tbParams = {
    ...
  }
  self.szMsg = tbParams[1]
  self.szNext = tbParams[2]
  self.pPanel:Label_SetText(self.txtContent2, self.szMsg)
  self:SynTimeInfo()
end

function tbWnd:SynTimeInfo()
  self.nContinueTime = GetTime() - Wedding.nStartTime
  self.nEndTime = Wedding.nStateEndTime - GetTime()
end

function tbWnd:ShowTime()
  self.nContinueTime = self.nContinueTime + 1
  self.pPanel:Label_SetText(self.txtContentTime1, Lib:TimeDesc3(self.nContinueTime))
  if self.nEndTime >= 1 then
    self.nEndTime = self.nEndTime - 1
  end
  self.pPanel:Label_SetText(self.txtContentTime, Lib:TimeDesc3(self.nEndTime))
  self.pPanel:Label_SetText(self.txtContentInfo1, i18n.Get(self.szNext))
  return true
end
