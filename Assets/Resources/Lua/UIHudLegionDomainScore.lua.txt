local tbWnd = Ui:CreateWindow("UIHudLegionDomainScore")
tbWnd.tbControls = {
  btnTong = "imgBG/DomainScoreInfo/btnList/btnTong",
  btnAttendee = "imgBG/DomainScoreInfo/btnList/btnIndividual",
  txtScoreTitle = "imgBG/DomainScoreInfo/ListTitle/txtMap",
  txtScoreValue = "imgBG/DomainScoreInfo/ListTitle/txtScore",
  goScoreTemplate = "imgBG/DomainScoreInfo/List/Viewport/Content/Template",
  goMyScore = "imgBG/DomainScoreInfo/MyScore",
  txtLegionTitle = "imgBG/DomainScoreInfo/btnList/btnTong/Text"
}
tbWnd.PAGE_INDEX = {TONG = 1, ATTENDEE = 2}

function tbWnd:NotifyEvents()
  return {
    {
      EventNotify.emNOTIFY_SYNC_LEGION_DOMAIN_SCORE,
      self.OnSyncDomainScore
    }
  }
end

function tbWnd:OnCreate()
  self:SetClickEmptyToClose()
  self.pPanel:Label_SetText(self.txtLegionTitle, i18n.EasyGet("\229\134\155\229\155\162"))
  self.tbPageBtnGroup = Ui:NewButtonGroup(self, {
    self.btnTong,
    self.btnAttendee
  }, true, self.SwitchPage)
  self.tbScoreScroll = Ui:NewScrollView(self, self.goScoreTemplate, nil, nil, true)
  self.tbDomainList = {}
  self.tbAttendeeList = {}
  self.nQueryDay = 0
end

function tbWnd:OnOpen()
  GlobalDomainBattle:QueryScoreReq()
  local nNowDay = Lib:GetLocalDay()
  if self.nQueryDay ~= nNowDay then
    self.tbDomainList = {}
    self.tbAttendeeList = {}
    self.nQueryDay = nNowDay
  end
  self.nPageIndex = nil
  self.tbPageBtnGroup:Select(self.PAGE_INDEX.TONG)
end

function tbWnd:SwitchPage(nIndex)
  if self.nPageIndex == nIndex then
    return
  end
  self.nPageIndex = nIndex
  self:UpdateShow()
end

function tbWnd:UpdateShow()
  self.tbScoreScroll:Reset()
  if self.nPageIndex == self.PAGE_INDEX.TONG then
    self.pPanel:Label_SetText(self.txtScoreTitle, i18n.EasyGet("\233\162\134\229\156\159"))
    self.pPanel:Label_SetText(self.txtScoreValue, i18n.EasyGet("\231\187\159\230\178\187\229\138\155"))
    for nIndex, tbInfo in pairs(self.tbDomainList) do
      self.tbScoreScroll:SetElement(nIndex, function(tbElement)
        local pPanel = tbElement.pPanel
        pPanel:Label_SetText("txtMap", Map:GetMapName(tbInfo.nMapTid))
        pPanel:Label_SetText("txtScore", tostring(tbInfo.nScore))
        pPanel:Label_SetText("txtRank", tostring(tbInfo.nRank))
      end)
    end
    self.pPanel:SetActive(self.goMyScore, false)
  elseif self.nPageIndex == self.PAGE_INDEX.ATTENDEE then
    self.pPanel:Label_SetText(self.txtScoreTitle, i18n.EasyGet("\229\144\141\229\173\151"))
    self.pPanel:Label_SetText(self.txtScoreValue, i18n.EasyGet("\231\167\175\229\136\134"))
    local nMyRank
    for nIndex, tbInfo in pairs(self.tbAttendeeList) do
      self.tbScoreScroll:SetElement(nIndex, function(tbElement)
        local pPanel = tbElement.pPanel
        pPanel:Label_SetText("txtMap", tbInfo.szName)
        pPanel:Label_SetText("txtScore", tostring(tbInfo.nScore))
        pPanel:Label_SetText("txtRank", tostring(nIndex))
      end)
      if me.szName == tbInfo.szName then
        nMyRank = nIndex
      end
    end
    if nMyRank then
      self.pPanel:SetActive(self.goMyScore, true)
      local tbMyInfo = self.tbAttendeeList[nMyRank]
      self.pPanel:Label_SetText(self.goMyScore .. "/txtName", tbMyInfo.szName)
      self.pPanel:Label_SetText(self.goMyScore .. "/txtScore", tbMyInfo.nScore)
      self.pPanel:Label_SetText(self.goMyScore .. "/txtRank", nMyRank)
    else
      self.pPanel:SetActive(self.goMyScore, false)
    end
  end
end

function tbWnd:OnSyncDomainScore(tbDomainList, tbAttendeeList)
  self.tbDomainList = tbDomainList
  self.tbAttendeeList = tbAttendeeList
  if Ui:WindowVisible(self.ClassName) then
    self:UpdateShow()
  end
end
