local tbWnd = Ui:CreateClass("UIHudLeftPanelBattle", "UIPanel")
tbWnd.tbControls = {
  txtRank = "MyRank/txtRank",
  txtScore = "MyRank/txtScore",
  txtTimer = "MyRank/txtTimer",
  txtTimerTips = "MyRank/txtTimerTips",
  txtRankTips = "MyRank/txtRankTips",
  txtScoreTips = "MyRank/txtScoreTips",
  txtFlagTips = "MyRank/txtFlagNumberTips",
  txtFlagNumber = "MyRank/txtFlagNumberTips/txtFlagNumber",
  txtCampTips = "MyRank/txtCampTips",
  txtCamp = "MyRank/txtCampTips/txtCamp"
}
tbWnd.tbOnClick = {}

function tbWnd:OnCreate()
end

function tbWnd:OnOpen()
end

function tbWnd:OnClose()
end

function tbWnd:OnActivate()
  if self.nTimerId then
    Timer:Close(self.nTimerId)
    self.nTimerId = nil
  end
  self.pPanel:Label_SetText(self.txtRank, Battle.nMyRank or 0)
  self.pPanel:Label_SetText(self.txtScore, Battle.tbSelfData and Battle.tbSelfData.nScore or 0)
  self.nEndTime = Battle.nEndTimeStamp and Battle.nEndTimeStamp - GetTime() or 0
  self.pPanel:Label_SetText(self.txtTimer, Lib:TimeDesc3(self.nEndTime))
  EventNotify:RegistNotify(EventNotify.emNOTIFY_SYNC_BATTLE_REPORT, self.OnSynData, self)
  EventNotify:RegistNotify(EventNotify.emNOTIFY_SYNC_BATTLE_TIME, self.OnSynTime, self)
  self.nTimerId = Timer:Register(Env.GAME_FPS, self.ShowTime, self)
end

function tbWnd:OnDeactivate()
  EventNotify:UnRegistNotify(EventNotify.emNOTIFY_SYNC_BATTLE_REPORT, self)
  EventNotify:UnRegistNotify(EventNotify.emNOTIFY_SYNC_BATTLE_TIME, self)
  if self.nTimerId then
    Timer:Close(self.nTimerId)
    self.nTimerId = nil
  end
end

function tbWnd:OnSynTime(nEndTimeStamp)
  self.nEndTime = nEndTimeStamp - GetTime()
  if not self.nTimerId then
    self.nTimerId = Timer:Register(Env.GAME_FPS, self.ShowTime, self)
  end
end

function tbWnd:OnSynData(tbSelfData, nMyRank, tbTeamScore, nLeftTime, nTeamIndex)
  if nLeftTime then
    self.nEndTime = nLeftTime
  end
  self.pPanel:Label_SetText(self.txtRank, nMyRank)
  self.pPanel:Label_SetText(self.txtScore, tbSelfData.nScore)
  self.pPanel:Label_SetText(self.txtCamp, Battle:GetTeamName(nTeamIndex))
  if not tbSelfData.nFlagCommitCount then
    self.pPanel:SetActive(self.txtFlagTips, false)
  else
    self.pPanel:SetActive(self.txtFlagTips, true)
    self.pPanel:Label_SetText(self.txtFlagNumber, tbSelfData.nFlagCommitCount)
  end
end

function tbWnd:ShowTime()
  self.nEndTime = self.nEndTime - 1
  if self.nEndTime <= 0 then
    self.nTimerId = nil
    return
  end
  self.pPanel:Label_SetText(self.txtTimer, Lib:TimeDesc3(self.nEndTime))
  return true
end
