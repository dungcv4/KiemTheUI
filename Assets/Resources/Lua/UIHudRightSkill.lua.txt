tbWnd.tbOnLongPressLoop = {
  btnSkill1 = function(self)
    if self.nSettingCastType == Ui.tbSettingType.tbCastTypeEnum.Action then
      local skillInfo = self.tbSkills[1]
      if skillInfo.CastType == FightSkill.SkillCasterType.castingaoe then
        local posx, posy = self.skillCaster:GetCastingPos()
        self:OnFinishCastAoe(1, posx, posy)
      end
      if skillInfo.CastType == FightSkill.SkillCasterType.castingdir then
        local dir = self.skillCaster:GetCastingDir()
        self:OnFinishCastDir(1, dir)
      end
      if skillInfo.CastType == FightSkill.SkillCasterType.none then
        self:onClickSkill(1)
      end
      return
    end
    self:onClickSkill(1)
  end,
  btnJump = function(self)
    PlayerQingKung:DoQingKung()
    TeamMgr:StopFollowMember()
  end

function tbWnd:OnSkillLongPressDown(nIdx)
  local skillInfo = self.tbSkills[nIdx]
  if not skillInfo then
    return
  end
  if self.nSettingCastType == Ui.tbSettingType.tbCastTypeEnum.Action then
    if me.GetSkillLevel(skillInfo.SkillId) <= 0 then
      return
    end
    local nAttackRadius = FightSkill.tbAllSkillSetting[skillInfo.SkillId].nAttackRadius
    local targetID = Operation:GetAttackTarget(skillInfo.SkillId)
    self.skillCaster:SetRadius(nAttackRadius)
    self.skillCaster:OpenManualSkillCaster(self.pPanel:GetObject(self[string.format("btnSkill%d", nIdx)]), skillInfo.CastType, targetID)
  elseif FightSkill:IsActiveSkill(nIdx) then
    local tbSkillInfo
    local nHRSSubType = FightSkill:GetHRSSubType(me)
    if FightSkill:IsDragonBoatSkillStatus(nHRSSubType) then
      tbSkillInfo = DragonBoatRace:GetPlayerOneSkillInfoWithoutConfig(nIdx)
    elseif FightSkill:IsKeYiMenSkillStatus(nHRSSubType) then
      tbSkillInfo = KeYiMen.Com:GetPlayerOneSkillInfoWithoutConfig(me, nIdx)
    elseif FightSkill:IsTongBattleSkillStatus(nHRSSubType) then
      tbSkillInfo = TongBattle:GetPlayerOneSkillInfoWithoutConfig(me, nIdx, nHRSSubType)
    elseif FightSkill.HRS_SUBTYPE_LIST.DynamicSkill == nHRSSubType then
      tbSkillInfo = FightSkill:GetDynamicPlayerOneSkillWithoutConfig(nIdx)
    elseif FightSkill.HRS_SUBTYPE_LIST.CustomSkill == nHRSSubType then
      return
    else
      tbSkillInfo = FightSkill:GetPlayerOneSkillInfoWithoutConfig(nIdx, me.nFactionSect)
    end
    Ui:OpenWindow("UISkillShow", tbSkillInfo, true, false)
  end
end

function tbWnd:OnSkillLongPressUp(nIdx)
  if self.nSettingCastType == Ui.tbSettingType.tbCastTypeEnum.Action then
    local skillInfo = self.tbSkills[nIdx]
    if skillInfo and me.GetSkillLevel(skillInfo.SkillId) > 0 then
      self.skillCaster:CloseManualSkillCaster(nIdx)
    end
  end
end

function tbWnd:OnManualCastFinish(ntype, ...)
  local param = {
    ...
  }
  if ntype == FightSkill.SkillCasterType.dir then
    self:OnFinishCastDir(param[1], param[2])
  elseif ntype == FightSkill.SkillCasterType.aoe then
    self:OnFinishCastAoe(param[1], param[2], param[3])
  end
end

function tbWnd:OnFinishCastDir(index, nDir)
  if me.nFightMode == Npc.FIGHT_MODE.emFightMode_None then
    me.CenterMsg(i18n.EasyGet("\229\174\137\229\133\168\229\140\186\229\134\133\228\184\141\232\131\189\228\189\191\231\148\168\232\175\165\230\138\128\232\131\189"))
    return
  end
  local nSkillID = self.tbSkills[index].SkillId
  Operation:SkillCheckHorseLimited(nSkillID)
  if me.GetSkillLevel(self.tbSkills[index].SkillId) <= 0 then
    return
  end
  Operation:UseSkillToDir(self.tbSkills[index].SkillId, nDir, 1)
end

function tbWnd:OnFinishCastAoe(index, nPosx, nPosy)
  if me.nFightMode == Npc.FIGHT_MODE.emFightMode_None then
    me.CenterMsg(i18n.EasyGet("\229\174\137\229\133\168\229\140\186\229\134\133\228\184\141\232\131\189\228\189\191\231\148\168\232\175\165\230\138\128\232\131\189"))
    return
  end
  local nSkillID = self.tbSkills[index].SkillId
  Operation:SkillCheckHorseLimited(nSkillID)
  if me.GetSkillLevel(self.tbSkills[index].SkillId) <= 0 then
    return
  end
  Operation:UseSkillToPos(self.tbSkills[index].SkillId, nPosx, nPosy)
end

function tbWnd:OnOpen()
  local nSubType = FightSkill:GetHRSSubType(me)
  local bFactionSkill = FightSkill:IsFactionSkillStatus(me, nSubType)
  local bCustomSkill = FightSkill:IsCustomSkillStatus(me, nSubType)
  self:UpdateMedicine(bFactionSkill, bCustomSkill)
  self:initSkill(nSubType)
  self:UpdateAFKState(nil, bFactionSkill, bCustomSkill)
  self:UpdateJumpStatus(bFactionSkill, bCustomSkill)
  for i = 1, SKILL_COUNT do
    local v = self.tbSkills[i]
    if v then
      self:UpdateSkillUsePoint(v.SkillId)
      self:OnUpdateSkillCD(v.SkillId)
    end
  end
  self:OnUpdateQingKungCD()
  self:UpdateQingKungUsePoint()
  self:OnUpdateMartialArtsCD()
  self:UpdateMartialArtsUsePoint()
  self:UpdateFightSkillBtnState()
  self.pPanel:SetActive(tbControls.skillCasterObj, false)
  self.nSettingCastType = Ui.PlayerPrefs.GetInt("SkillCastType", 1)
  self:ChangeLongPressTimeBaseOnCastType()
  self:UpdateStealStateLock()
  self:UpdateShowTips()
  self:UpdateSwitchTargetState(bFactionSkill, bCustomSkill)
end

function tbWnd:initDynamicSkill()

    tbSkillInfo = FightSkill.tbAllSkillSetting[nSkillID]
    if nSkillID > 0 and tbSkillInfo then
      tbSkill = {

      self.tbUseSkill[nSkillID] = tbSkill
      self.tbSkills[nIndex] = tbSkill
      self.pPanel:Sprite_SetSprite(tbSkillPathList[tbSkill.SortInUI], tbSkill.szIconAltlas)
      self.pPanel:SetActive(self.tbSkillButtonPath[nIndex], true)
    else
      self.tbUseSkill[nSkillID] = nil
      self.tbSkills[nIndex] = nil
      self.pPanel:Sprite_SetSprite(tbSkillPathList[nIndex], "")
      self.pPanel:SetActive(self.tbSkillButtonPath[nIndex], false)
    end

function tbWnd:initSkill(nSubType)
  for _, szButtonKey in pairs(self.tbSkillButtonPath) do
    self.pPanel:SetActive(szButtonKey, true)
  end
  nSubType = nSubType or FightSkill:GetHRSSubType(me)
  if FightSkill:IsDragonBoatSkillStatus(nSubType) then
    self:initDragonBoatSkill()
    return
  end
  if FightSkill:IsKeYiMenSkillStatus(nSubType) then
    self:initKeYiMenSkill()
    return
  end
  if FightSkill:IsTongBattleSkillStatus(nSubType) then
    self:initTongBattleSkill(nSubType)
    return
  end
  if nSubType == FightSkill.HRS_SUBTYPE_LIST.DynamicSkill then
    self:initDynamicSkill()
    return
  end
  if nSubType == FightSkill.HRS_SUBTYPE_LIST.CustomSkill then
    self:initCustomSkill()
    return
  end
  self.tbSkills = {}
  self.tbUseSkill = {}
  local tbShowSkill = FightSkill:GetFactionSkill(me.nFaction, me.nFactionSect)
  if tbShowSkill then
    for k, skill in pairs(tbShowSkill) do
      local v = Lib:CopyTB(skill)
      self.tbUseSkill[v.SkillId] = v
      self.tbSkills[v.SortInUI] = v
      if 1 == v.SortInUI then
        self.pPanel:Sprite_SetSprite(tbControls.imgSkillIcon1, v.IconAltlas)
      elseif 2 == v.SortInUI then
        self.pPanel:Sprite_SetSprite(tbControls.imgSkillIcon2, v.IconAltlas)
      elseif 3 == v.SortInUI then
        self.pPanel:Sprite_SetSprite(tbControls.imgSkillIcon3, v.IconAltlas)
      elseif 4 == v.SortInUI then
        self.pPanel:Sprite_SetSprite(tbControls.imgSkillIcon4, v.IconAltlas)
      elseif 5 == v.SortInUI then
        self.pPanel:Sprite_SetSprite(tbControls.imgSkillIcon5, v.IconAltlas)
      end
      if v.SortInUI <= SKILL_COUNT and v.SortInUI >= 1 then
        self:UpdateSkillUsePoint(v.SkillId)
        self:OnUpdateSkillCD(v.SkillId)
      end
    end
  end
  self:UpdateFightSkillBtnState()
  self:RefreshHotkeyDisplay()
  self.tbQingKungInfo = self.tbQingKungInfo or {}
  self:OnUpdateQingKungCD()
  self:UpdateQingKungUsePoint()
  self:UpdateStealStateLock()
  local tbWuXueSkillInfo = WuXue:GetAllWuXueInstalled(me)
  if tbWuXueSkillInfo and tbWuXueSkillInfo[1] and tbWuXueSkillInfo[1] > 0 then
    local nWuXueSkillId = WuXue:GetSkillIdByWuXueId(tbWuXueSkillInfo[1])
    self.pPanel:SetActive(tbControls.btnSkillMartialArts, true)
    self.pPanel:Sprite_SetSprite(tbControls.imgSkillIconMartialArts, FightSkill:GetSkillIconAtlas(nWuXueSkillId))
  else
    self.pPanel:SetActive(tbControls.btnSkillMartialArts, false)
  end
end

function tbWnd:onClickSkill(index)
  EventNotify.OnNotify(EventNotify.emNOTIFY_TRY_USE_SKILL, index)
  if me.nFightMode == Npc.FIGHT_MODE.emFightMode_None then
    me.CenterMsg(i18n.EasyGet("\229\174\137\229\133\168\229\140\186\229\134\133\228\184\141\232\131\189\228\189\191\231\148\168\232\175\165\230\138\128\232\131\189"))
    return
  end
  local bEnable = false
  if index > 0 and index <= SKILL_COUNT then
    local skillinfo = self.tbSkills[index]
    if skillinfo then
      local nSkillLevel = me.GetSkillLevel(skillinfo.SkillId)
      if nSkillLevel > 0 then
        bEnable = true
      end
    end
  end
  if not bEnable then
    if FightSkill:IsSpecialSkillStatus(FightSkill:GetHRSSubType(me)) then
      me.CenterMsg(i18n.EasyGet("\229\189\147\229\137\141\230\156\170\230\139\165\230\156\137\232\175\165\230\138\128\232\131\189"))
      return
    end
    me.CenterMsg(i18n.EasyGet("\230\138\128\232\131\189\231\173\137\231\186\167\228\184\1860 \239\188\140\228\184\141\232\131\189\228\189\191\231\148\168"))
    return
  end
  EventNotify.OnNotify(EventNotify.emNOTIFY_USE_SKILL)
  Operation:Attack(self.tbSkills[index].SkillId, false)
end

function tbWnd:UpdateMedicine(bFactionSkill, bCustomSkill)
  if nil == bFactionSkill then
    bFactionSkill = FightSkill:IsFactionSkillStatus(me)
  end
  if nil == bCustomSkill then
    bCustomSkill = FightSkill:IsCustomSkillStatus(me)
  end
  if not bFactionSkill and not bCustomSkill then
    self.pPanel:SetActive(tbControls.btnMedicine, false)
    return
  end
  local bHasMedicine = MedicineMgr:HasMedicine()

function tbWnd:UpdateSkillCD(nSkillID, bNotify)
  local pNpc = me.GetNpc()
  if not pNpc then
    return
  end
  local tbInfo = self.tbUseSkill[nSkillID]
  if not tbInfo then
    return
  end
  local nSkillPos = tbInfo.SortInUI
  local nNextFrame, nCurCDTime = pNpc.GetSkillNextCastTime(nSkillID)
  if nNextFrame <= 0 then
    tbCacheSkillTime[nSkillID] = nil
    if self.tbSkillCD and nSkillPos and self.tbSkillCD[nSkillPos] then
      self.tbSkillCD[nSkillPos]:Reset()
    end
    return
  end
  local nLogicFrame = GetFrame()
  local nFrame = nNextFrame - nLogicFrame
  if nFrame < 0 then
    nFrame = 0
  end
  local fTotalCDTime = nCurCDTime / Env.GAME_FPS
  local fTime = nFrame / Env.GAME_FPS
  local fTotalTime = fTotalCDTime > 0 and fTotalCDTime or fTime
  if tbInfo.bUseTotalTime then
    fTotalTime = tbInfo.fTotalTime
  end
  local nCdfTime = fTime
  local nCdfTotalTime = fTotalTime
  local bShowLable = true
  local nStartFrame = tbCacheSkillTime[nSkillID]
  if fTotalTime < 1 or not nStartFrame then
    tbCacheSkillTime[nSkillID] = nil
    nCdfTime = 0
    nCdfTotalTime = 0
  else
    nCdfTotalTime = fTotalCDTime
  end
  local bIsUsePoint, nUsePoint, nPerUsePoint = pNpc.GetSkillUsePoint(nSkillID)
  if self.tbSkillCD and nSkillPos and self.tbSkillCD[nSkillPos] then
    self.tbSkillCD[nSkillPos]:Run(bIsUsePoint, nUsePoint, nPerUsePoint, nCdfTime, nCdfTotalTime)
  end
  tbInfo.fTotalTime = fTotalTime
end

function tbWnd:UpdateFightSkillBtnState()
  local bInFightMode = me.nFightMode ~= Npc.FIGHT_MODE.emFightMode_None
  for i = 1, SKILL_COUNT do
    local skillinfo = self.tbSkills[i]
    local bDisable = true
    if bInFightMode and skillinfo then
      local nSkillLevel = me.GetSkillLevel(skillinfo.SkillId)
      if nSkillLevel > 0 then
        bDisable = false
      end
    end
    local key = string.format("imgBG/btnSkill%d/DisableMask", i)
    self.pPanel:SetActive(key, bDisable)
    local particals = string.format("imgBG/btnSkill%d/anniutishi", i)
    self.pPanel:SetActive(particals, not bDisable)
    if FightSkill:IsActiveSkill(i) and skillinfo then
      local bLevelEnoughLearn = me.nLevel >= (skillinfo.GainLevel or 0)
      local szLockIconPath = string.format("imgBG/btnSkill%d/SkillLock", i)
      self.pPanel:SetActive(szLockIconPath, not bLevelEnoughLearn)
      if not bLevelEnoughLearn then
        local szLockTextPath = string.format("imgBG/btnSkill%d/SkillLock/txtUnlockLv", i)
        self.pPanel:Label_SetText(szLockTextPath, i18n.Get(FightSkill.Terms.LevelRequire, skillinfo.GainLevel))
      end
    end
  end
  self.pPanel:SetActive(tbControls.btnSkillMartialArts .. "/DisableMask", not bInFightMode)
end

function tbWnd:UpdateMartialArtsCD()

    if self.tbSkillMartialArtsCD then

    end
    return
  end

    self.nMartialArtsCDCacheTime = nil
    if self.tbSkillMartialArtsCD then
      self.tbSkillMartialArtsCD:Reset()
    end
    return
  end
  local nLogicFrame = GetFrame()
  local nFrame = nNextFrame - nLogicFrame
  if nFrame < 0 then
    nFrame = 0
  end
  local fTotalCDTime = nCurCDTime / Env.GAME_FPS
  local fTime = nFrame / Env.GAME_FPS
  local fTotalTime = fTotalCDTime > 0 and fTotalCDTime or fTime
  local nCdfTime = fTime
  local nCdfTotalTime = fTotalTime
  local nStartFrame = self.nMartialArtsCDCacheTime
  if fTotalTime < 1 or not nStartFrame then
    self.nMartialArtsCDCacheTime = nil
    nCdfTime = 0
    nCdfTotalTime = 0
  else
    nCdfTotalTime = fTotalCDTime
  end
  local bIsUsePoint, nUsePoint, nPerUsePoint = pNpc.GetSkillUsePoint(nActiveSkillId)
  if self.tbSkillMartialArtsCD then
    self.tbSkillMartialArtsCD:Run(bIsUsePoint, nUsePoint, nPerUsePoint, nCdfTime, nCdfTotalTime)
  end
end
