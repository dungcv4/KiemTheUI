end

function tbWnd:OnCreate()
  Ui:CloseWindow("UILoginChannelInner")
  Ui:CloseWindow("UILoginChannelSDK")
  self.FactionView = Ui:NewScrollView(self, tbControls.FactionTemplate, self._SelectFaction)
  self.FactionView:Clear()
  local tbFactionList = {}
  for nFactionId = 1, Faction.MAX_FACTION_COUNT do
    if Faction:IsFactionOpen(nFactionId) then
      table.insert(tbFactionList, nFactionId)
    end
  end
  self.tbFaction2Idx = {}
  local tbCurFactionList = {}
  for nKey, _ in pairs(tbSerisControls) do
    for nIndex = 1, #tbFactionList do
      local nFactionId = tbFactionList[nIndex]
      local nSeries = Faction:GetSeries(nFactionId)
      if nKey == nSeries then
        table.insert(tbCurFactionList, nFactionId)
        self.tbFaction2Idx[nFactionId] = #tbCurFactionList
      end
    end
  end
  for nIndex = 1, #tbCurFactionList do
    self.FactionView:SetElement(nIndex, function(tbElement)
      tbElement.nFactionId = tbCurFactionList[nIndex]
      local pPanel = tbElement.pPanel
      pPanel:Sprite_SetSprite("imgIcon", Faction:GetIcon(tbElement.nFactionId))
    end)
  end
  local nCurFactionIndex = MathRandom(#tbCurFactionList)
  local nCurFactionId = tbCurFactionList[nCurFactionIndex]
  local tbSexList = {}
  if Faction:IsOpenSex(nCurFactionId, Npc.SexDef.MALE) then
    table.insert(tbSexList, Npc.SexDef.MALE)
  end
  if Faction:IsOpenSex(nCurFactionId, Npc.SexDef.FEMALE) then
    table.insert(tbSexList, Npc.SexDef.FEMALE)
  end
  local nSex = Lib:RandTable(tbSexList)
  self:SetSex(nSex)
  self.nCurFactionId = nCurFactionId
  self.pPanel:SetActive(tbControls.btnArrowUp, false)
  self.pPanel:SetActive(tbControls.btnArrowDown, true)
  self.pPanel:SetActive(tbControls.SeriesList, true)
end

function tbWnd:OnClose()
  Ui:ResetCamera()
end

function tbWnd:OnOpen(tbRoleList)
  self.loginflag = false
  self.tbRoleList = tbRoleList
  self:SetSex(self.nCurSelectSex)
  self.FactionView:SelectElement(self.tbFaction2Idx[self.nCurFactionId])
  local nServerId = Sdk:GetServerId()
  local tbSerList = GetServerList()
  for i, v in ipairs(tbSerList) do
    if nServerId == v.dwServerId then
      Login.ServerName = v.szName
      break
    end
  end
  self:ChooseNewFaction()
  self:onClickRandomName(self.nCurSelectSex)
end

function tbWnd:ChooseNewFaction()
  for nFactionId = Faction.MAX_FACTION_COUNT, 1, -1 do
    if Faction:IsFactionOpen(nFactionId) then
      local nCurSeries = Faction:GetSeries(nFactionId)
      local tbSexList = {}
      if Faction:IsOpenSex(nFactionId, Npc.SexDef.MALE) then
        table.insert(tbSexList, Npc.SexDef.MALE)
      end
      if Faction:IsOpenSex(nFactionId, Npc.SexDef.FEMALE) then
        table.insert(tbSexList, Npc.SexDef.FEMALE)
      end
      local nSex = Lib:RandTable(tbSexList)
      self:SetSex(nSex)
      self.nCurFactionId = nFactionId
      local nSex = Lib:RandTable(tbSexList)
      self:SetSex(nSex)
      self.nCurFactionId = nFactionId
      for nSeries, szButton in ipairs(self.tbSeriesButton) do
        self.pPanel:Toggle_SetChecked(szButton, nCurSeries == nSeries)
      end
      self.FactionView:SelectElement(self.tbFaction2Idx[self.nCurFactionId])
      break
    end
  end
end

function tbWnd:onClickRandomName(nSex)
  local nRand = MathRandom(self.nAllWeight)
  local szName = Login:GetRandomName(nSex, nRand <= self.nHasOtherWeight)
  for i = 1, 3 do
    if CheckNameAvailable(szName) then
      break
    end
  end
  self.pPanel:Input_SetText(tbControls.inputCreateRoleName, szName)
end

function tbWnd:SetSex(nSex)
  self.nCurSelectSex = nSex
  if nSex == Npc.SexDef.MALE then
    self.pPanel:Button_SetEnable(tbControls.btnMale2, false)
    self.pPanel:Button_SetEnable(tbControls.btnFemale2, true)
    self.pPanel:Toggle_SetEnable(tbControls.btnSexMale, false)
    self.pPanel:Toggle_SetEnable(tbControls.btnSexFemale, true)
    self.pPanel:ToggleGroup_SetSelect(tbControls.btnSexMale, true)
    self.pPanel:ToggleGroup_SetSelect(tbControls.btnSexFemale, false)
  elseif nSex == Npc.SexDef.FEMALE then
    self.pPanel:Button_SetEnable(tbControls.btnFemale2, false)
    self.pPanel:Button_SetEnable(tbControls.btnMale2, true)
    self.pPanel:Toggle_SetEnable(tbControls.btnSexMale, true)
    self.pPanel:Toggle_SetEnable(tbControls.btnSexFemale, false)
    self.pPanel:ToggleGroup_SetSelect(tbControls.btnSexFemale, true)
    self.pPanel:ToggleGroup_SetSelect(tbControls.btnSexMale, false)
  end
end

function tbWnd:SelectSex(nSex)
  self:SetSex(nSex)
  EventNotify.OnNotify(EventNotify.emNOTIFY_UILOGINBG_ACTION, "Select", self.nCurFactionId, nSex)
end

function tbWnd:OnCreateRoleRespond(nCode, nRoleID)
  Ui:CloseWindow("UILoadingTips")
  if 0 ~= nCode then
    self.bCreated = false
  else
    self.loginflag = true
    LoginRole(nRoleID)
    Ui.PlayerPrefs.SetInt("LoginRoleID", nRoleID)
    EventNotify.OnNotify(EventNotify.emNOTIFY_CREATE_ROLE_SUCCESS, nRoleID, self.nCurFactionId, self.nCurSelectSex)
  end
end

function tbWnd:OnSelectSeries(nSeries)
  local tbFactionList = {}
  for i = 1, self.FactionView:GetCount() do
    local tbElement = self.FactionView:GetElement(i)
    local bActive = Faction:GetSeries(tbElement.nFactionId) == nSeries
    tbElement.pPanel:SetActive("Main", bActive)
    if bActive then
      table.insert(tbFactionList, i)
    end
  end
  if Faction:GetSeries(self.nCurFactionId) ~= nSeries then
    self.FactionView:SelectElement(Lib:RandTable(tbFactionList))
  end
end

function tbWnd:UnSelectSeries()
end

function tbWnd:_SelectFaction(_, tbElement)
  local nFactionId = tbElement.nFactionId
  self.nCurFactionId = nFactionId
  local bOpenMale = Faction:IsOpenSex(self.nCurFactionId, Npc.SexDef.MALE)
  local bOpenFemale = Faction:IsOpenSex(self.nCurFactionId, Npc.SexDef.FEMALE)
  self.pPanel:Toggle_SetEnable(tbControls.btnSexMale, bOpenMale)
  self.pPanel:Toggle_SetEnable(tbControls.btnSexFemale, bOpenFemale)
  EventNotify.OnNotify(EventNotify.emNOTIFY_UILOGINBG_ACTION, "PlayDuo", nFactionId)
  if not bOpenMale and self.nCurSelectSex == Npc.SexDef.MALE then
    self:SelectSex(Npc.SexDef.FEMALE)
  elseif not bOpenFemale and self.nCurSelectSex == Npc.SexDef.FEMALE then
    self:SelectSex(Npc.SexDef.MALE)
  else
    self:SelectSex(self.nCurSelectSex)
  end
  self.pPanel:SetActive(tbControls.btnSexMale, bOpenMale)
  self.pPanel:SetActive(tbControls.btnMale2, bOpenMale)
  self.pPanel:SetActive(tbControls.btnSexFemale, bOpenFemale)
  self.pPanel:SetActive(tbControls.btnFemale2, bOpenFemale)
  self.pPanel:Label_SetText(tbControls.txtFactionDesc, Faction:GetDesc(nFactionId))
  self.pPanel:Sprite_SetSprite(tbControls.imgFactionInfo, Faction:GetRadarChartPath(tbElement.nFactionId))
  local szName = Faction:GetName(nFactionId)
  local nSeries = Faction:GetSeries(nFactionId)
  for nKey, tb in pairs(tbSerisControls) do
    if nKey == nSeries then
      self.pPanel:Label_SetText(tb[2], szName)
      self.pPanel:SetActive(tb[1], true)
    else
      self.pPanel:SetActive(tb[1], false)
    end
  end
  for nKey, szPath in pairs(tbTxtSeriesControls) do
    if nKey == nSeries then
      self.pPanel:SetActive(szPath, true)
    else
      self.pPanel:SetActive(szPath, false)
    end
  end
end

function tbWnd:OnMapLoaded()
  self:Close()
end

function tbWnd:OnServerConnectLost()
  if Ui.bForRetrunLogin then
    return
  end
  Ui:OpenWindow("UIMessageBoxBig", i18n.Get("DisconnectedFromServer"), {
    function()
      Ui:BackToLogin(true)
    end
  }, {
    i18n.szOk
  }, nil, nil, true)
end

function tbWnd:CloseByDefault()
  Ui.bForRetrunLogin = true
  Ui:ReturnToLogin()
  CloseServerConnect()
  Ui.bForRetrunLogin = nil
end

function tbWnd:OnLoginRoleRespond(bSuccessed, nCode, dwRoleId)
  Ui:OnLoginRoleRespond(bSuccessed, nCode, dwRoleId)
end
