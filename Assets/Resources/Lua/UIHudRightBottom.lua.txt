function tbWnd:OnCreate()
  self.pPanel:SetActive(tbControls.EffectTemplate, false)
  self.pPanel:SetActive(tbControls.btnAchievement, true)
  self.pPanel:SetActive(tbControls.btnWuLinRongYu, true)
  self.nCountDownFreeTimer = nil
end

function tbWnd:OnOpen()
  self:UpdateFightState()
  self:RegistRedPoint()
  self:UpdateIcon(false)
  self:SetIconsShow()
  if HanWuYiJi:CheckInHanWuYiJi() then
    self:OnJoinHanWuYiJi()
  end
  if DragonBoatRace:CheckInDragonBoatRace() then
    self:OnJoinDragonBoatRace()
  end
end

function tbWnd:OnClose()
  if self.nCountDownFreeTimer then
    Timer:Close(self.nCountDownFreeTimer)
    self.nCountDownTimer = nil
  end
  if self.nCountDownCoinFreeTimer then
    Timer:Close(self.nCountDownCoinFreeTimer)
    self.nCountDownCoinFreeTimer = nil
  end
  EventNotify.OnNotify(EventNotify.emNOTIFY_ON_CLOSE_HUD_RIGHT_BOTTOM)
  self:UnRegistRedPoint()
end

function tbWnd:SetBagEntranceRedPointState(bState)
  local objRedPoint = self.pPanel:GetObject(tbControls.imgBag .. "/RedPoint")
  if objRedPoint then
    objRedPoint:SetActive(bState)
  end
end

function tbWnd:onClickSwitchPanel()
  self.pPanel:SetActive(tbControls.panelSystem, self.isShowSystemPanel)
  if self.isShowSystemPanel then
    Ui:CloseWindow("UIHudRightSkill")
    self.pPanel:SetActive("imgBG/btnSwitch/imgSwitchSkill", false)
  else
    Ui:OpenWindow("UIHudRightSkill")
    self.pPanel:SetActive("imgBG/btnSwitch/imgSwitchSkill", true)
  end
  self.isShowSystemPanel = not self.isShowSystemPanel
  EventNotify.OnNotify(EventNotify.emNOTIFY_TRY_SWITCH_RIGHT_HUD_PANEL, self.isShowSystemPanel)
end

function tbWnd:OnChangeFightState()
  self:UpdateFightState()
end

function tbWnd:UpdateFightState()
  if Ui.bInZone then
    self.pPanel:SetActive(tbControls.panelSystem, false)
    Ui:OpenWindow("UIHudRightSkill")
    self.pPanel:SetActive("imgBG/btnSwitch/imgSwitchSkill", true)
    return
  end
  if 0 == me.nFightMode then
    self.isShowSystemPanel = true
  else
    self.isShowSystemPanel = false
  end
  self:onClickSwitchPanel()
end

function tbWnd:SetPanel(bIsShowSystemPanel)
  self.isShowSystemPanel = bIsShowSystemPanel
  self:onClickSwitchPanel()
end

function tbWnd:NotifyEvents()
  return {
    {
      EventNotify.emNOTIFY_CHANGE_FIGHT_STATE,
      self.OnChangeFightState
    },
    {
      EventNotify.emNOTIFY_CHANGE_PLAYER_LEVEL,
      self.OnLevelChange
    },
    {
      EventNotify.emNOTIFY_LOGIN_ZONE,
      self.OnLoginZone
    },
    {
      EventNotify.emNOTIFY_LOGOUT_ZONE,
      self.OnLogoutZone
    }
  }
end

function tbWnd:OnLevelChange()
  self:UpdateIcon(true)
end

function tbWnd:UpdateIcon(bIsLevelChangeCall)
  if Ui.bInZone then
    return
  end
  for _, tbABtnLevelInfo in ipairs(Ui.UIHudRightBottom.tbBtnLevelCfg) do
    local szButtonKey = self[tbABtnLevelInfo.szBtnName]
    self.pPanel:SetActive(szButtonKey, me.nLevel >= tbABtnLevelInfo.nLevel)
    if bIsLevelChangeCall and me.nLevel == tbABtnLevelInfo.nLevel then
      local _, _, szMatch = string.find(szButtonKey, "(/[^/]+/[^/]+)$")
      local szIconRelativeKey = string.sub(szMatch, 2, string.len(szMatch)) .. "/img"
      local szEffectKey = szButtonKey .. "/img/Hud_Effect"
      if self.pPanel.transform:Find(szEffectKey) == nil then
        self.pPanel:CloneObject(tbControls.EffectTemplate, "Hud_Effect", szIconRelativeKey)
      else
        self.pPanel:SetActive(szEffectKey, true)
      end
    end
  end
end

function tbWnd:OnClickCloseEffect(szButtonKey)
  local szEffectKey = self[szButtonKey] .. "/img/Hud_Effect"
  if self.pPanel.transform:Find(szEffectKey) ~= nil then
    self.pPanel:SetActive(szEffectKey, false)
  end
end

function tbWnd:SetIconsShow()
  local tbBtnLevel = {}
  for _, tbInfo in ipairs(Ui.UIHudRightBottom.tbBtnLevelCfg) do
    tbBtnLevel[tbInfo.szBtnName] = tbInfo.nLevel
  end
  local bShow = not Ui.bInZone
  local tbBtns = {
    btnLifeSkill = tbControls.btnLifeSkill,
    btnAchievement = tbControls.btnAchievement,
    btnHonorLevel = tbControls.btnHonorLevel,
    btnKin = tbControls.btnKin,
    btnTong = tbControls.btnTong,
    btnRanking = tbControls.btnRanking,
    btnPartner = tbControls.btnPartner,
    btnWuLinRongYu = tbControls.btnWuLinRongYu
  }
  local nLevel = me.nLevel
  for szName, szPath in pairs(tbBtns) do
    self.pPanel:SetActive(szPath, bShow and (not tbBtnLevel[szName] or nLevel >= tbBtnLevel[szName]))
  end
end

function tbWnd:OnLoginZone()
  if HanWuYiJi:CheckInHanWuYiJi() then
    self:OnJoinHanWuYiJi()
  end
  if DragonBoatRace:CheckInDragonBoatRace() then
    self:OnJoinDragonBoatRace()
  end
  self:SetIconsShow()
end

function tbWnd:OnLogoutZone()
  self.pPanel:SetActive(tbControls.btnBag, true)
  self.pPanel:SetActive(tbControls.btnSkill, true)
  self:SetIconsShow()
end

function tbWnd:OnJoinHanWuYiJi()
  self.pPanel:SetActive(tbControls.btnBag, false)
  self.pPanel:SetActive(tbControls.btnSkill, false)
  self.pPanel:SetActive(tbControls.btnLifeSkill, false)
  self.pPanel:SetActive(tbControls.btnKin, false)
  self.pPanel:SetActive(tbControls.btnTong, false)
  self.pPanel:SetActive(tbControls.btnRanking, false)
  self.pPanel:SetActive(tbControls.btnPartner, false)
  self.pPanel:SetActive(tbControls.btnHonorLevel, false)
  self.pPanel:SetActive(tbControls.btnAchievement, false)
  self.pPanel:SetActive(tbControls.btnWuLinRongYu, false)
end

function tbWnd:OnJoinDragonBoatRace()
  self.pPanel:SetActive(tbControls.btnBag, false)
  self.pPanel:SetActive(tbControls.btnSkill, false)
end
