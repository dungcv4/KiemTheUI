function tbWnd:OnDlcGiftChange()
  local isAllDownload = true
  local nDLCTatalSize = Client.UpdateModule.GetDLCTotalSize()
  local nNeedDownloadDLCSize = Client.UpdateModule.GetNeedDownDLCSize()
  if nDLCTatalSize > 0 and nNeedDownloadDLCSize > 0 then
    self.pPanel:SetActive(tbControls.btnCTA, true)
  end
  if nNeedDownloadDLCSize > 0 then
    isAllDownload = false
    self.nProgressTimer = Timer:Register(Env.GAME_FPS * 5, self.CheckDlcProcess, self)
  end
  if Gift.tbDlcAwardData and not Gift:IsGetAllDLCGift() then
    self.pPanel:SetActive(tbControls.btnCTA, true)
    if isAllDownload then
      RedPointMgr:LightRedPoint(RedPointMgr.EVENT.DlcAward)
    end
  end
end

function tbWnd:CheckDlcProcess()
  local nDLCTatalSize = Client.UpdateModule.GetDLCTotalSize()
  local nReadDownLoadSize = Client.UpdateModule.GetCurDownloadedDLCSize()
  local nSavedSize = nReadDownLoadSize + Client.UpdateModule.GetDownloadedDLCSize()
  local nPercent = 1
  if 0 ~= nDLCTatalSize then
    nPercent = nSavedSize / nDLCTatalSize
    nPercent = 0.2 + nPercent * 0.8
  end
  self.nCurPercent = nPercent
  self.nCurIndex = self.nCurIndex or 0
  if not Gift:IsGetAllDLCGift() then
    if self.nCurPercent >= 0.2 and 0 == self.nCurIndex then
      RedPointMgr:LightRedPoint(RedPointMgr.EVENT.DlcAward)
      self.nCurIndex = 1
    end
    if self.nCurPercent >= 0.6 and 1 == self.nCurIndex then
      RedPointMgr:LightRedPoint(RedPointMgr.EVENT.DlcAward)
      self.nCurIndex = 2
    end
    if self.nCurPercent >= 1 and 2 == self.nCurIndex then
      RedPointMgr:LightRedPoint(RedPointMgr.EVENT.DlcAward)
      self.nCurIndex = 3
    end
  end
  if self.nCurPercent >= 1 then
    if Gift:IsGetAllDLCGift() then
      self.pPanel:SetActive(tbControls.btnCTA, false)
    end
    return false
  else
    return true
  end
end

function tbWnd:OnDlcGiftChangeButton()
  local nDLCTatalSize = Client.UpdateModule.GetDLCTotalSize()
  local nReadDownLoadSize = Client.UpdateModule.GetCurDownloadedDLCSize()
  local nSavedSize = nReadDownLoadSize + Client.UpdateModule.GetDownloadedDLCSize()
  local nPercent = 1
  if 0 ~= nDLCTatalSize then
    nPercent = nSavedSize / nDLCTatalSize
  end
  if nPercent >= 1 and Gift:IsGetAllDLCGift() then
    self.pPanel:SetActive(tbControls.btnCTA, false)
  end
end

function tbWnd:UpdateIcon(bIsLevelChangeCall)
  local bIsInFuBen = Map:IsFuben(me.nMapTemplateId)
  if bIsInFuBen then
    for _, tbABtnLevelInfo in ipairs(Ui.UIHudRightTopWelfare.tbBtnShowCfg) do
      local szButtonKey = tbControls[tbABtnLevelInfo.szBtnName]
      local bNormalShow = true
      if szBtnCheckShowFn[tbABtnLevelInfo.szBtnName] then
        bNormalShow = szBtnCheckShowFn[tbABtnLevelInfo.szBtnName]()
      end
      local bFuBenShow = 1 == tbABtnLevelInfo.nIsShowInFuBen
      local nShowLevel = tbABtnLevelInfo.nLevel
      if tbABtnLevelInfo.szBtnName == "btnShop" and (not OnOffController:Check(OnOffController.KEY.ShopLevelLimit) or not nShowLevel) then
        nShowLevel = 1
      end
      local bLevelEnoughShow = 0 == tbABtnLevelInfo.nIsLvControl or nShowLevel <= me.nLevel
      self.pPanel:SetActive(szButtonKey, bFuBenShow and bLevelEnoughShow and bNormalShow)
    end
    return
  end
  for _, tbABtnLevelInfo in ipairs(Ui.UIHudRightTopWelfare.tbBtnShowCfg) do
    if 1 == tbABtnLevelInfo.nIsLvControl then
      local szButtonKey = tbControls[tbABtnLevelInfo.szBtnName]
      local bShow = true
      if szBtnCheckShowFn[tbABtnLevelInfo.szBtnName] then
        bShow = szBtnCheckShowFn[tbABtnLevelInfo.szBtnName]()
      end
      local nShowLevel = tbABtnLevelInfo.nLevel
      if tbABtnLevelInfo.szBtnName == "btnShop" and (not OnOffController:Check(OnOffController.KEY.ShopLevelLimit) or not nShowLevel) then
        nShowLevel = 1
      end
      bShow = nShowLevel <= me.nLevel and bShow
      self.pPanel:SetActive(szButtonKey, bShow)
      if bIsLevelChangeCall and me.nLevel == tbABtnLevelInfo.nLevel then
        local _, _, szMatch = string.find(szButtonKey, "(/[^/]+/[^/]+)$")
        local szIconRelativeKey = string.sub(szMatch, 2, string.len(szMatch)) .. "/imgIcon"
        local szEffectKey = szButtonKey .. "/imgIcon/Hud_Effect"
        if self.pPanel.transform:Find(szEffectKey) == nil then
          self.pPanel:CloneObject(self.EffectTemplate, "Hud_Effect", szIconRelativeKey)
        else
          self.pPanel:SetActive(szEffectKey, true)
        end
      end
    end
  end
  self:FirstRechargeBtnShow()
  self:LevelUpBtnShow()
end

    LocalData:SetRoleNumber("ShopRedPoint", Shop.Cfg.DefaultShopRedPointValue)

  end
end

function tbWnd:FirstRechargeBtnShow()
  self.pPanel:SetActive(tbControls.btnFirstRecharge, false)
  if self.bIsInFuBen == true then
    return
  end
  if me.nLevel < Recharge.nFirstRechargeJoinLevel then
    return
  end
  local nGetFirst = me.GetUserValue(Recharge.FIRST_RECHARGE_GROUP, Recharge.KEY_GET_FIRST_RECHARGE)
  if 1 ~= nGetFirst then
    self.pPanel:SetActive(tbControls.btnFirstRecharge, true)
    if FirstRecharge.bHadOpenUIFirstRecharge == false then
      RedPointMgr:LightRedPoint(RedPointMgr.EVENT.FirstRecharge)
    end
    return
  end
  self.pPanel:SetActive(tbControls.FirstRechargeEffect, 1 ~= nGetFirst)
  local nAward = me.GetUserValue(Recharge.FIRST_RECHARGE_GROUP, Recharge.KEY_GET_AWARD)
  if nAward == Recharge.tbFirstRechargeDayBit.MAX - 1 then
    self.pPanel:SetActive(tbControls.btnFirstRecharge, false)
    return
  end
  self.pPanel:SetActive(tbControls.btnFirstRecharge, true)
  local nTime = me.GetUserValue(Recharge.FIRST_RECHARGE_GROUP, Recharge.KEY_FIRST_RECHARGE_TIME)
  local nDayOffset = Lib:CalcDayOffset(14400, nTime)
  for i = 1, nDayOffset + 1 do
    if 0 == bit32.band(nAward, Recharge.tbFirstRechargeDayBit[i]) then
      RedPointMgr:LightRedPoint(RedPointMgr.EVENT.FirstRecharge)
      break
    end
  end
end

function tbWnd:BtnTimeLimitShow()
  local tbCurrentGiftInfo = LimitedTimeGift:GetAllGiftInfo()
  if not tbCurrentGiftInfo then
    self.pPanel:SetActive(tbControls.btnTimeLimit, false)
    return
  end
  if self.bIsInFuBen == true then
    return
  end
  local nLongestLeftTime = 0
  for _, tbTypeGift in pairs(tbCurrentGiftInfo) do
    for _, tbInfo in pairs(tbTypeGift) do
      if tbInfo and tbInfo.nEndTime and nLongestLeftTime < tbInfo.nEndTime then
        nLongestLeftTime = tbInfo.nEndTime
      end
    end
  end
  if nLongestLeftTime <= 0 then
    self.pPanel:SetActive(tbControls.btnTimeLimit, false)
    return
  end
  local nCurTime = GetTime()
  local nLeftTime = nLongestLeftTime - nCurTime
  if nLeftTime <= 0 then
    self.pPanel:SetActive(tbControls.btnTimeLimit, false)
  else
    self.pPanel:Label_SetText(tbControls.btnTimeLimit .. "/Text", LimitedTimeGift:TimeDesc(nLeftTime))
    self.pPanel:SetActive(tbControls.btnTimeLimit, true)
  end
end

function tbWnd:DealActivityShow(tbActivityCfg, szBtnPath)
  local nCurTime = GetTime()
  local nStartTime = Lib:ParseConfigDateTime(tbActivityCfg.szStartTime)
  local nEndTime = Lib:ParseConfigDateTime(tbActivityCfg.szEndTime)
  local bIsInFuBen = Map:IsFuben(me.nMapTemplateId)
  local bShow = not bIsInFuBen and me.nLevel >= tbActivityCfg.nLevel
  bShow = bShow and nCurTime >= nStartTime and nCurTime < nEndTime
  self.pPanel:SetActive(szBtnPath, bShow)
  self.pPanel:Label_SetText(szBtnPath .. "/Text", i18n.Get(tbActivityCfg.szIconName))
end

function tbWnd:DealSummerGiftShow()
  local bIsInFuBen = Map:IsFuben(me.nMapTemplateId)
  local bShow = not bIsInFuBen and not Ui.bInZone
  if not bShow then
    self.pPanel:SetActive(tbControls.btnSummerGift, bShow)
    return
  end
  local pInstance = ActivityMgr:GetInstanceByClassName(ActivityMgr.Def.ClassNameList.CelebrationGift)
  if not (pInstance and pInstance:IsOpen()) or not pInstance:CanJoin(me.nLevel) then
    bShow = false
  end
  self.pPanel:SetActive(tbControls.btnSummerGift, bShow)
  if bShow then
    local tbConfig = pInstance:GetConfig()
    self.pPanel:Label_SetText(tbControls.btnSummerGift .. "/Text", i18n.Parse(pInstance:GetName()))
    self.pPanel:Sprite_SetSprite(tbControls.btnSummerGift .. "/imgIcon", tbConfig.szImgIcon, true)
  end
end

function tbWnd:DealEntranceSchedule()
  self.tbEntranceSV:Reset()
  local nCurTime = GetTime()
  local nStartTime, nEndTime, bShow
  local nMapTemplateId = me.nMapTemplateId
  local bIsInFuBen = Map:IsFuben(nMapTemplateId) and not Map:IsZoneHangUpMap(nMapTemplateId)
  for i, tbCfg in pairs(ActivityMgr.EntranceScheduleCfg) do
    nStartTime = Lib:ParseConfigDateTime(tbCfg.szStartTime)
    nEndTime = Lib:ParseConfigDateTime(tbCfg.szEndTime)
    bShow = not bIsInFuBen and me.nLevel >= tbCfg.nLevel
    bShow = bShow and nCurTime >= nStartTime and nCurTime < nEndTime
    if bShow and tbCfg.szCheckShowFunc then
      bShow = ActivityMgr.EntranceSchedule:CallScript(tbCfg.szCheckShowFunc, table.unpack(tbCfg.CheckShowFuncParams or {}))
    end
    if bShow then
      self.tbEntranceSV:SetElement(i, function(tbElement)
        tbElement.nCfgId = i
        tbElement.pPanel:Label_SetText("Text", i18n.Get(tbCfg.szIconName))
        tbElement.pPanel:Sprite_SetSprite("imgIcon", tbCfg.szIconImg)
        if tbCfg.RedPointList then
          RedPointMgr:DynamicRegisteRedPoint(self.pPanel, tbElement.szKey, self:DealRedPointList(tbCfg.RedPointList))
        end
        if tbCfg.bUseEffect then
          tbElement.pPanel:SetActive("imgIcon/Hud_Effect", tbCfg.bUseEffect)
        end
      end)
      if tbCfg.szCheckShowRedPointFunc then
        ActivityMgr.EntranceSchedule:CallScript(tbCfg.szCheckShowRedPointFunc, table.unpack(tbCfg.CheckShowRedPointFuncParams or {}))
      end
    end
  end
end

function tbWnd:DealRedPointList(tbRedPointList)
  self.tbEntranceRedPointList = self.tbEntranceRedPointList or {}
  for k, _ in pairs(self.tbEntranceRedPointList) do
    self.tbEntranceRedPointList[k] = nil
  end
  for k, v in ipairs(tbRedPointList) do
    self.tbEntranceRedPointList[k] = RedPointMgr.EVENT[v]
  end
  return self.tbEntranceRedPointList
end

  end
end
