local tbWnd = Ui:CreateWindow("UIHudHanWuYiJi")

function tbWnd:NotifyEvents()
  return {
    {
      EventNotify.emNOTIFY_SYNC_ITEM,
      self.RefreshView,
      self
    },
    {
      EventNotify.emNOTIFY_DEL_ITEM,
      self.RefreshView,
      self
    }
  }
end

local tbControls = {
  btnRole = "imgBG/BtnList/btnRole",
  btnEnhance = "imgBG/BtnList/btnEnhance",
  btnGive = "imgBG/Bag/Fushi/btnGive",
  wndFushi = "wndFushi",
  UIItemGrid = "imgBG/Bag/Fushi/UIItemGrid",
  ItemTemplate = "imgBG/Bag/ItemList/ItemTemplate",
  txtFuShiNum = "imgBG/Bag/Fushi/imgNumBg/txtNum",
  PlayerTemplate = "wndFushi/imgBg/PlayerList/PlayerTemplate",
  btnSkill = "imgBG/BtnList/btnSkill",
  btnGiveClose = "wndFushi/imgBg/btnClose",
  btnHide = "imgBG/btnHandle",
  btnShow = "imgBG/btnHandle2",
  Bag = "imgBG/Bag"
}
tbWnd.tbControls = tbControls
tbWnd.tbOnClick = {
  btnRole = function(self)
    self:OpenRoleView()
  end,
  btnEnhance = function(self)
    self:OpenEnhanceView()
  end,
  btnGive = function(self)
    self:OpenGiveView()
  end,
  btnSkill = function(self)
    self:OpenSkillView()
  end,
  btnGiveClose = function(self)
    self:CloseGiveView()
  end,
  btnHide = function(self)
    self:HideBag()
  end,
  btnShow = function(self)
    self:ShowBag()
  end
}

function tbWnd:OnCreate()
  self.tbItemGrid = Ui:NewExistItemGrid(self, "imgBG/Bag/Fushi/UIItemGrid")
  self.pItemScorllView = Ui:NewScrollView(self, self.ItemTemplate, function(_, nIndex, tbElement)
  end, false, false)
  self.pFushiGiveScorllView = Ui:NewScrollView(self, self.PlayerTemplate, function(_, nIndex, tbElement)
  end, false, false)
end

function tbWnd:OnOpen()
  self.nMinInputCount = 0
  self.pPanel:SetActive(tbControls.wndFushi, false)
  self.pPanel:SetActive(tbControls.btnHide, true)
  self.pPanel:SetActive(tbControls.btnShow, false)
  self.pPanel:SetActive(tbControls.Bag, true)
  self.pPanel:SetActive(tbControls.btnRole, true)
  self:RefreshView()
  RedPointMgr:DynamicRegisteRedPoint(self.pPanel, tbControls.btnEnhance, {
    RedPointMgr.EVENT.HanWuEnhance
  })
end

function tbWnd:OnClose()
  RedPointMgr:DynamicUnRegist(self.pPanel, tbControls.btnEnhance)
end

function tbWnd:RefreshView()
  self.nMaxInputCount = me.GetItemCountByTypeInBag(HanWuYiJi.FuShiTemplatedId)
  HanWuYiJi:RefreshEnhanceRedPoint()
  self:UpdateFuShiGrid()
  self:InitItemScorllView()
  self:InitFushiGiveScorllView()
end

function tbWnd:UpdateFuShiGrid()
  local nCount = me.GetItemCountByTypeInBag(HanWuYiJi.FuShiTemplatedId)
  self.tbItemGrid:SetGenericItemTemplate(HanWuYiJi.FuShiTemplatedId, nCount)
  self.pPanel:Label_SetText(self.txtFuShiNum, nCount)
  self.tbItemGrid:SetClickEnable(true)
end

function tbWnd:OpenRoleView()
  Ui:OpenWindow("UIBag", 2)
end

function tbWnd:OpenEnhanceView()
  Ui:OpenWindow("UIHanWuEnhance")
end

function tbWnd:OpenGiveView()
  self.pPanel:SetActive(tbControls.wndFushi, true)
end

function tbWnd:CloseGiveView()
  self.pPanel:SetActive(tbControls.wndFushi, false)
end

function tbWnd:InitItemScorllView()
  self.pItemScorllView:Clear()
  local tbItem = me.GetItemListInBag()
  local tbShowItemList = {}
  for i, pItem in ipairs(tbItem) do
    local bIsExclude = false
    for i, nItemTempateId in ipairs(HanWuYiJi.EXCLUDE_ITEM) do
      if nItemTempateId == pItem.dwTemplateId then
        bIsExclude = true
        break
      end
    end
    if false == bIsExclude then
      table.insert(tbShowItemList, pItem)
    end
  end
  local nIndex = 1
  local nMaxBagCount = HanWuYiJi.MaxBagCount
  while nIndex <= nMaxBagCount do
    local tbHanWuBagItemData = {}
    local pItem = tbShowItemList[nIndex]
    if nil ~= pItem then
      tbHanWuBagItemData.bIsEmpty = false
      tbHanWuBagItemData.pItem = pItem
    else
      tbHanWuBagItemData.bIsEmpty = true
    end
    self.pItemScorllView:SetElement(nIndex, function(tbElement)
      self:SetItemElementInfo(tbElement, tbHanWuBagItemData)
    end)
    nIndex = nIndex + 1
  end
end

function tbWnd:SetItemElementInfo(tbElement, tbHanWuBagItemData)
  if tbHanWuBagItemData.bIsEmpty == true then
    tbElement.pPanel:SetActive("UIItemGrid", false)
  else
    tbElement.pPanel:SetActive("UIItemGrid", true)
    local pItem = tbHanWuBagItemData.pItem
    local tbItemGrid = tbElement.tbItemGrid
    tbItemGrid = tbItemGrid or Ui:NewExistItemGrid(tbElement, "UIItemGrid")
    local nCount = me.GetItemCountByTypeInBag(pItem.dwTemplateId)
    tbItemGrid:SetItem(pItem.dwId, pItem.dwTemplateId)
    tbItemGrid.szItemOpt = "HanWu_Item"
    tbItemGrid:SetClickEnable(true)
  end
end

function tbWnd:InitFushiGiveScorllView()
  self.pFushiGiveScorllView:Clear()
  local tbMemberList = HanWuYiJi:GetMemberList()
  local nIndex = 1
  for nMemberId, tbMemberData in pairs(tbMemberList) do
    self.pFushiGiveScorllView:SetElement(nIndex, function(tbElement)
      self:SetPlayerElementInfo(tbElement, tbMemberData, nMemberId)
    end)
    nIndex = nIndex + 1
  end
end

function tbWnd:SetPlayerElementInfo(tbElement, tbMemberData, nMemberId)
  tbElement.pPanel:Sprite_SetSprite("PortraitObj/Portrait/Image", PlayerPortrait:GetIcon(tbMemberData.nPortrait))
  PlayerPortraitFrame:SetPortraitFrame(tbElement.pPanel, "PortraitObj/PortraitFrame", tbMemberData.nPortraitFrame)
  tbElement.pPanel:Label_SetText("PortraitObj/Level/Text", tbMemberData.nLevel)
  tbElement.pPanel:Label_SetText("txtPlayerName", tbMemberData.szName)
  tbElement.AddAndSubGroup = Ui:NewExistAddAndSubGroup(tbElement, "UIAddAndSubGroup")
  tbElement.AddAndSubGroup:BindEvent(function(nValue)
    tbElement.nGiveCount = self:UpdateNumberInput(nValue)
    return tbElement.nGiveCount
  end)
  tbElement.AddAndSubGroup:SetNumRange(self.nMinInputCount, self.nMaxInputCount)
  tbElement.AddAndSubGroup:SetInputValue(tbElement.nGiveCount or 0)
  Ui:Button_BindEvent(tbElement.pPanel, "btnGive", function()
    self:OnClickGiveFuShiBtn(tbElement.nGiveCount or 0, nMemberId)
  end)
end

function tbWnd:UpdateNumberInput(nNum)
  if nNum < self.nMinInputCount then
    nNum = self.nMinInputCount
    return nNum
  end
  if nNum > self.nMaxInputCount then
    nNum = self.nMaxInputCount
    return nNum
  end
  return nNum
end

function tbWnd:OnClickGiveFuShiBtn(nCount, nBeGivePlayerId)
  HanWuYiJi:OnGiveFuShi(nCount, nBeGivePlayerId)
end

function tbWnd:OpenSkillView()
  Ui:OpenWindow("UISkill", 1, false, true)
end

function tbWnd:OnUseItem(tbHanWuBagItemData)
  if tbHanWuBagItemData.bIsEmpty == true then
    return
  else
    local pItem = tbHanWuBagItemData.pItem
    Item:ClientUseItem(pItem.dwId)
  end
end

function tbWnd:HideBag()
  self.pPanel:SetActive(tbControls.wndFushi, false)
  self.pPanel:SetActive(tbControls.Bag, false)
  self.pPanel:SetActive(tbControls.btnHide, false)
  self.pPanel:SetActive(tbControls.btnShow, true)
end

function tbWnd:ShowBag()
  self.pPanel:SetActive(tbControls.wndFushi, false)
  self.pPanel:SetActive(tbControls.Bag, true)
  self.pPanel:SetActive(tbControls.btnHide, true)
  self.pPanel:SetActive(tbControls.btnShow, false)
end
