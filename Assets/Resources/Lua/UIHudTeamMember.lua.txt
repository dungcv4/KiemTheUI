local UIHudTeamMember = Ui:CreateClass("UIHudTeamMember", "UIPanel")
local tbControls = {
  PortraitObj = "PortraitObj",
  txtRoleName = "Info/txtRoleName",
  hpBar = "Info/hpBg/hp",
  imgLeader = "imgLeader",
  txtFollow = "txtFollow",
  imgSelected = "imgSelected",
  imgPractice = "imgPractice"
}
UIHudTeamMember.tbControls = tbControls

function UIHudTeamMember:OnCreate(nIndex)
  self.tbPortraitObj = Ui:NewExistPortraitObjPanel(self, tbControls.PortraitObj)
  self.followPanel = self.followPanel or {}
  self.pPanel:SetActive("txtFollow", false)
end

function UIHudTeamMember:UpdateDeathState()
  self.pPanel:Label_SetText("txtFollow", i18n.EasyGet("\229\183\178\229\135\186\229\177\128"))
  self.pPanel:SetActive("txtFollow", true)
end

function UIHudTeamMember:UpdateFollowState(tbMemberData)
  local tbMeBeFollowed = TeamMgr:GetMeBeFollowed()
  local nTargetId = TeamMgr:GetFollowTarget(me.dwID)
  local nMemberTargetId = TeamMgr:GetFollowTarget(tbMemberData.nPlayerID)
  if nTargetId and tbMemberData.nPlayerID == nTargetId then
    self.pPanel:Label_SetText("txtFollow", i18n.Get(TeamMgr.Terms.FollowHe))
    self.pPanel:SetActive("txtFollow", true)
    return
  end
  if nMemberTargetId and nMemberTargetId == me.dwID then
    self.pPanel:Label_SetText("txtFollow", i18n.Get(TeamMgr.Terms.FollowMe))
    self.pPanel:SetActive("txtFollow", true)
    return
  end
  self.pPanel:SetActive("txtFollow", false)
end

function UIHudTeamMember:SetMainActive(bActive)
  self.pPanel:SetActive("Main", bActive)
end

function UIHudTeamMember:SetMainPanelData(tbMemberData, bDeath)
  if not tbMemberData then
    return
  end
  self.tbPortraitObj:SetRoleIcon(tbMemberData.nPortrait)
  self.tbPortraitObj:SetFactionIcon(tbMemberData.nFaction)
  self.tbPortraitObj:SetLevel(tbMemberData.nLevel)
  self.tbPortraitObj:SetPortraitFrame(tbMemberData.nPortraitFrameID or 0)
  self.pPanel:Label_SetText(tbControls.txtRoleName, tbMemberData.szName)
  self.pPanel:SetActive(tbControls.imgLeader, TeamMgr:IsCaptain(tbMemberData.nPlayerID))
  self.pPanel:Sprite_SetFill(tbControls.hpBar, tbMemberData.nHpPercent / 100)
  if bDeath then
    self:UpdateDeathState()
  else
    self:UpdateFollowState(tbMemberData)
  end
end

function UIHudTeamMember:SetFakePanelData(tbAssistNpcInfo, tbCaptainData)
  self.tbPortraitObj:SetRoleIcon(tbAssistNpcInfo.nPortrait)
  self.tbPortraitObj:SetFactionIcon(tbAssistNpcInfo.nFaction)
  self.pPanel:Label_SetText(tbControls.txtRoleName, i18n.Parse(KNpc.GetNameByTemplateId(tbAssistNpcInfo.nNpcTemplateId)))
  self.tbPortraitObj:SetLevel(tbCaptainData.nLevel)
  self.pPanel:SetActive(tbControls.imgLeader, false)
  self.pPanel:Sprite_SetFill(tbControls.hpBar, 1.0)
end

function UIHudTeamMember:SetImgPractice(bActive)
  self.pPanel:SetActive(tbControls.imgPractice, bActive)
end
