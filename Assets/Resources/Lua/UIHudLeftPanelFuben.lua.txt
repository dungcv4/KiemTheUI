local tbWnd = Ui:CreateClass("UIHudLeftPanelFuben", "UIPanel")
tbWnd.tbControls = {
  UIDefaultFubenInfo = "UIDefaultFubenInfo",
  UIGoldBossOutputInfo = "UIGoldBossOutputInfo",
  UIHudLeftPanelDungeon = "UIHudLeftPanelDungeon",
  UIHudLeftPanelBattle = "UIHudLeftPanelBattle",
  UIDuanWuOutputInfo = "UIDuanWuOutputInfo",
  UISnowFight = "UISnowFight",
  PanelGuard = "PanelGuard",
  PanelFunnyPlay = "PanelBeachBrawl",
  btnScore = "btnScore",
  btnScore1 = "btnScore1"
}
tbWnd.tbOnClick = {
  btnScore = function(self)
    self:OnClickBtnScore()
  end,
  btnScore1 = function(self)
    local bRet, tbFunc = FubenUIController:IsNeedFuncScoreButton()
    if bRet and tbFunc then
      Lib:CallBack(tbFunc)
    end
  end
}
tbWnd.tbMutexPanel = {
  {
    Name = "UIDefaultFubenInfo",
    Obj = tbWnd.tbControls.UIDefaultFubenInfo
  },
  {
    Name = "UIGoldBossOutputInfo",
    Obj = tbWnd.tbControls.UIGoldBossOutputInfo
  },
  {
    Name = "UIHudLeftPanelDungeon",
    Obj = tbWnd.tbControls.UIHudLeftPanelDungeon
  },
  {
    Name = "UIHudLeftPanelBattle",
    Obj = tbWnd.tbControls.UIHudLeftPanelBattle
  },
  {
    Name = "UIDuanWuOutputInfo",
    Obj = tbWnd.tbControls.UIDuanWuOutputInfo
  },
  {
    Name = "UISnowFight",
    Obj = tbWnd.tbControls.UISnowFight
  },
  {
    Name = "UIHudLeftPanelWedding",
    Obj = tbWnd.tbControls.PanelGuard
  },
  {
    Name = "UIHudLeftPanelFunnyPlay",
    Obj = tbWnd.tbControls.PanelFunnyPlay
  }
}
tbWnd.CurrentPanel = tbWnd.CurrentPanel or "None"

function tbWnd:SetCurrentPanel(szName)
  tbWnd.CurrentPanel = szName
end

function tbWnd:GetCurrentPanel()
  return tbWnd.CurrentPanel
end

function tbWnd:Init()
  for i = 1, #tbWnd.tbMutexPanel do
    local ref = tbWnd.tbMutexPanel[i]
    self.pPanel:SetActive(ref.Obj, false)
  end
end

function tbWnd:OnCreate()
  self.Panel = {}
  for i = 1, #self.tbMutexPanel do
    local ref = self.tbMutexPanel[i]
    self.Panel[ref.Name] = {}
    self.Panel[ref.Name].Info = self.tbMutexPanel[i]
    self.Panel[ref.Name].pPanel = Ui:CreatePanel(ref.Name, ref.Obj, self)
  end
  self:Init()
end

function tbWnd:OnActivate()
  local subPanel = self:GetSubPanel(tbWnd:GetCurrentPanel())
  if subPanel and subPanel.OnActivate then
    subPanel:OnActivate(self.szFubenType)
  end
  if FubenUIController:IsShowBtnScore() then
    self.pPanel:SetActive(self.btnScore, true)
  else
    self.pPanel:SetActive(self.btnScore, false)
  end
end

function tbWnd:OnDeactivate()
  local subPanel = self:GetSubPanel(tbWnd:GetCurrentPanel())
  if subPanel and subPanel.OnDeactivate then
    subPanel:OnDeactivate(self.szFubenType)
  end
end

function tbWnd:OnOpen()
  local subPanel = self:GetSubPanel(tbWnd:GetCurrentPanel())
  if subPanel and subPanel.OnOpen then
    subPanel:OnOpen(self.szFubenType)
  end
end

function tbWnd:OnClose()
  local subPanel = self:GetSubPanel(tbWnd:GetCurrentPanel())
  if subPanel and subPanel.OnClose then
    subPanel:OnClose(self.szFubenType)
  end
end

function tbWnd:UpdateData(szFuben, ...)
  local subPanel = self:GetSubPanel(szFuben)
  if subPanel and subPanel.UpdateData then
    subPanel:UpdateData(...)
  end
end

function tbWnd:UpdateContentData(szFuben, ...)
  local subPanel = self:GetSubPanel(szFuben)
  if subPanel and subPanel.UpdateFubenInfo then
    subPanel:UpdateFubenInfo(...)
  end
end

function tbWnd:UpdateTimerData(szFuben, ...)
  local subPanel = self:GetSubPanel(szFuben)
  if subPanel and subPanel.UpdateTimerData then
    subPanel:UpdateTimerData(...)
  end
end

function tbWnd:GetSubPanel(szName)
  if not self.Panel[szName] then
    return nil
  end
  return self.Panel[szName].pPanel
end

function tbWnd:GetSubPanelInfo(szName)
  if not self.Panel[szName] then
    return nil
  end
  return self.Panel[szName].Info
end

function tbWnd:HidePanel(szPanelName)
  local currentPanel = self:GetSubPanel(tbWnd:GetCurrentPanel())
  if not currentPanel then
    return
  end
  if szPanelName == tbWnd:GetCurrentPanel() then
    if currentPanel and currentPanel.OnDeactivate then
      currentPanel:OnDeactivate()
    end
    self.pPanel:SetActive(self:GetSubPanelInfo(tbWnd:GetCurrentPanel()).Obj, false)
  end
end

function tbWnd:ShowPanel(szPanelName)
  local newPanel = self:GetSubPanel(szPanelName)
  if not newPanel then
    print(debug.traceback())
    return
  end
  tbWnd:SetCurrentPanel(szPanelName)
  self.pPanel:SetActive(self:GetSubPanelInfo(szPanelName).Obj, true)
  if newPanel.OnActivate then
    newPanel:OnActivate(self.szFubenType)
  end
end

function tbWnd:SwitchPanel(szPanelName)
  self:HidePanel(tbWnd:GetCurrentPanel())
  self:ShowPanel(szPanelName)
  self:OnChangeFuncScoreButton()
end

function tbWnd:UpdateBtnScoreStatus()
  if FubenUIController:IsShowBtnScore() then
    self.pPanel:SetActive(self.btnScore, true)
  else
    self.pPanel:SetActive(self.btnScore, false)
  end
end

function tbWnd:OnClickBtnScore()
  if "UIHudLeftPanelBattle" == tbWnd:GetCurrentPanel() then
    Ui:OpenWindow("UIBattleRank", false, true)
    return
  end
  if Ui:WindowVisible("UIHudOutputInfo") then
    Ui:GetWindow("UIHudOutputInfo"):Close()
  elseif TongChallenge:IsInTongChallengePlaying() then
    Ui:OpenWindow("UIHudOutputInfo", "TongChallenge")
  elseif GuardXiangYangMgr:IsInGuardXiangYang() then
    Ui:OpenWindow("UIHudOutputInfo", "GuardXiangYang")
  end
end

function tbWnd:OnChangeFuncScoreButton()
  local bRet = FubenUIController:IsNeedFuncScoreButton()
  if bRet then
    self.pPanel:SetActive(self.btnScore1, true)
  else
    self.pPanel:SetActive(self.btnScore1, false)
  end
end
