local tbWnd = Ui:CreateClass("UIHudLeftPanelDungeon", "UIPanel")
tbWnd.tbControls = {
  challengeDescObj = "Root/ChallengeMsg",
  successConditionTxt = "Root/ChallengeMsg/SuccessConditionTxt",
  progressTitle = "Root/ChallengeMsg/Progress/Title",
  progressTxt = "Root/ChallengeMsg/Progress/Txt",
  remainTimeDesc = "Root/ChallengeMsg/RemainTimeTxt",
  introTxt = "Root/ChallengeMsg/IntroTxt"
}
tbWnd.tbOnClick = {}

function tbWnd:OnCreate()
end

function tbWnd:OnOpen()
  if FubenUIController.nReceiveTime then
    self:UpdateFubenInfo()
  else
    self.pPanel:SetActive(self.successConditionTxt, false)
    self.pPanel:SetActive(self.remainTimeDesc, false)
    self.pPanel:SetActive(self.introTxt, false)
    self.pPanel:Label_SetText(self.progressTxt, i18n.EasyGet("<color=#{0}>\231\173\137\229\190\133\229\137\175\230\156\172\229\188\128\229\167\139</color>", Ui.tbColor:GetColorByType("Intro")))
    self.pPanel:SetActive(self.progressTxt, true)
    self.pPanel:SetActive(self.progressTitle, true)
  end
end

function tbWnd:OnClose()
  if self.nTimer ~= nil then
    Timer:Close(self.nTimer)
    self.nTimer = nil
  end
end

function tbWnd:OnActivate()
  self:OnOpen()
end

function tbWnd:OnDeactivate()
  self:OnClose()
end

function tbWnd:UpdateData()
  self:UpdateFubenInfo()
end

function tbWnd:UpdateFubenInfo()
  local szColor = Ui.tbColor:GetColorByType("Intro")
  if FubenUIController.szCondition then
    local szDesc = string.format("<color=#%s>", szColor) .. i18n.Parse(FubenUIController.szCondition) .. "</color>"
    self.pPanel:Label_SetText(self.successConditionTxt, i18n.EasyGet("<color=#{1}>\232\191\135\229\133\179\230\157\161\228\187\182:</color> {0}", szDesc, Ui.tbColor:GetColorByType("Classify")))
    self.pPanel:SetActive(self.successConditionTxt, true)
  else
    self.pPanel:SetActive(self.successConditionTxt, false)
  end
  if FubenUIController.szProgress then
    local szDesc = string.format("<color=#%s>", szColor) .. i18n.Parse(FubenUIController.szProgress) .. "</color>"
    self.pPanel:Label_SetText(self.progressTxt, szDesc)
    self.pPanel:SetActive(self.progressTxt, true)
    self.pPanel:SetActive(self.progressTitle, true)
  else
    self.pPanel:SetActive(self.progressTxt, false)
    self.pPanel:SetActive(self.progressTitle, false)
  end
  self.tbEndTime = {}
  if FubenUIController.tbLastTime then
    for index, varTime in pairs(FubenUIController.tbLastTime) do
      if type(varTime) == "table" then
        self.tbEndTime[index] = {
          varTime[1],
          varTime[2] + FubenUIController.nReceiveTime
        }
      else
        self.tbEndTime[index] = varTime + FubenUIController.nReceiveTime
      end
    end
  end
  if self.nTimer ~= nil then
    Timer:Close(self.nTimer)
    self.nTimer = nil
  end
  if not Lib:IsEmptyTB(self.tbEndTime) then
    self.pPanel:Label_SetText(self.remainTimeDesc, "")
    self.pPanel:SetActive(self.remainTimeDesc, true)
    self.nTimer = Timer:Register(Env.GAME_FPS, self.UpdateLeftTime, self)
    self:UpdateLeftTime()
  else
    self.pPanel:SetActive(self.remainTimeDesc, false)
  end
  if FubenUIController.szIntro then
    self.pPanel:Label_SetText(self.introTxt, i18n.Parse(FubenUIController.szIntro))
    self.pPanel:SetActive(self.introTxt, true)
  else
    self.pPanel:SetActive(self.introTxt, false)
  end
end

function tbWnd:UpdateLeftTime()
  local tbDesc = {}
  local nNowTime = GetTime()
  local szColor = Ui.tbColor:GetColorByType("ClassifyParameter")
  for _, varTime in pairs(self.tbEndTime) do
    if type(varTime) == "table" then
      local szDesc, nEndTime = table.unpack(varTime)
      local nLeftTime = nEndTime - nNowTime
      if nLeftTime >= 0 then
        local szTime = string.format("<color=#%s>", szColor) .. Lib:TimeDesc(nLeftTime) .. "</color>"
        local szTextColor = Ui.tbColor:GetColorByType("Classify")
        table.insert(tbDesc, string.format("%s\239\188\154%s", string.format("<color=#%s>", szTextColor) .. i18n.Parse(szDesc) .. "</color>", szTime))
      end
    else
      local nLeftTime = varTime - nNowTime
      if nLeftTime >= 0 then
        local szDesc = string.format("<color=#%s>", szColor) .. Lib:TimeDesc(nLeftTime) .. "</color>"
        table.insert(tbDesc, i18n.EasyGet("<color=#{1}>\229\137\169\228\189\153\230\151\182\233\151\180\239\188\154</color>{0}", szDesc, Ui.tbColor:GetColorByType("Classify")))
      end
    end
  end
  if not Lib:IsEmptyTB(tbDesc) then
    self.pPanel:Label_SetText(self.remainTimeDesc, table.concat(tbDesc, "\n"))
    return true
  else
    self.nTimer = nil
    return false
  end
end
