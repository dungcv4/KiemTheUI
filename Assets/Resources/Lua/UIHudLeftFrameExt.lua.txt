function tbWnd:OnCreate()
  tbWnd.Panel = {}
  for i = 1, #tbWnd.PanelInfo do
    local ref = tbWnd.PanelInfo[i]
    tbWnd.Panel[ref.Name] = {}
    tbWnd.Panel[ref.Name].Info = tbWnd.PanelInfo[i]
    tbWnd.Panel[ref.Name].pPanel = Ui:CreatePanel(ref.Name, ref.Obj, self)
  end
  self.nBattleRoyaleHonor = 0
  for key, value in ipairs(WuLinGaoShou.tbAllTMapSetting) do
    tbWnd.tbOnMapLoadFunc[key] = function()
      RemoteServer.GoldBossReqUI()
    end
  end
  self:UpdateRedPacketButton(false)
end

tbWnd.panelTaskTeam = 1
tbWnd.panelTaskTeamFuBen = 2

function tbWnd:SetPanelState(nPanelControl, nPanelSelected)
  if nPanelControl and self.nPanelState == nPanelControl then
    return
  end
  if nPanelSelected and nPanelSelected == self.CurrentPanel then
    return
  end
  nPanelControl = nPanelControl or self.panelTaskTeam
  nPanelSelected = nPanelSelected or self.TP_TASK
  if nPanelControl == self.panelTaskTeam then
    self.pPanel:SetActive(tbControls.ButtonGroup, true)
    self.pPanel:SetActive(tbControls.ButtonGroup2, false)
    for i = 1, #tbWnd.PanelInfo do
      local ref = tbWnd.PanelInfo[i]
      if nPanelSelected == ref.bit then
        if self.pPanel:Toggle_GetChecked(ref.Toggle) == false then
          self.pPanel:Toggle_SetChecked(ref.Toggle, true)
        end
        self.pPanel:SetActive(ref.Obj, true)
        self:GetSubPanel(ref.Name):OnActivate()
      else
        self.pPanel:Toggle_SetChecked(ref.Toggle, false)
        self.pPanel:SetActive(ref.Obj, false)
        self:GetSubPanel(ref.Name):OnDeactivate()
      end
    end
  else
    self.pPanel:SetActive(tbControls.ButtonGroup, false)
    self.pPanel:SetActive(tbControls.ButtonGroup2, true)
    for i = 1, #tbWnd.PanelInfo do
      local ref = tbWnd.PanelInfo[i]
      if nPanelSelected == ref.bit then
        if false == self.pPanel:Toggle_GetChecked(ref.Toggle2) then
          self.pPanel:Toggle_SetChecked(ref.Toggle2, true)
        end
        self.pPanel:SetActive(ref.Obj, true)
        self:GetSubPanel(ref.Name):OnActivate()
      else
        self.pPanel:Toggle_SetChecked(ref.Toggle2, false)
        self.pPanel:SetActive(ref.Obj, false)
        self:GetSubPanel(ref.Name):OnDeactivate()
      end
    end
  end
  self.CurrentPanel = nPanelSelected
  self.nPanelState = nPanelControl
end

function tbWnd:OnToggleFuben(isOn, nPanelType)
  if not isOn then
    self:GetSubPanel("UIHudLeftPanelFuben"):OnDeactivate()
    self.pPanel:SetActive(tbControls.FubenPanelPath, false)
    return
  end
  if self.CurrentPanel == tbWnd.TP_FUBEN and self.nPanelState == nPanelType then
    return
  end
  self.CurrentPanel = tbWnd.TP_FUBEN
  self.pPanel:SetActive(tbControls.FubenPanelPath, true)
  if self:GetSubPanel("UIHudLeftPanelFuben").OnActivate then
    self:GetSubPanel("UIHudLeftPanelFuben"):OnActivate()
  end
end

function tbWnd:OnToggleTask(isOn, nPanelType)
  if not isOn then
    self:GetSubPanel("UIHudLeftPanelTask"):OnDeactivate()
    self.pPanel:SetActive(tbControls.TaskPanelPath, false)
    return
  end
  if self.CurrentPanel == tbWnd.TP_TASK and self.nPanelState == nPanelType then
    Ui:OpenWindow("UITask")
    return
  end
  self.CurrentPanel = tbWnd.TP_TASK
  self.pPanel:SetActive(tbControls.TeamPanelPath, false)
  self.pPanel:SetActive(tbControls.TaskPanelPath, true)
  self.pPanel:SetActive(tbControls.FubenPanelPath, false)
  if self:GetSubPanel("UIHudLeftPanelTask").OnActivate then
    self:GetSubPanel("UIHudLeftPanelTask"):OnActivate()
  end
end

function tbWnd:OnToggleTeam(isOn, nPanelType)
  if not isOn then
    self:GetSubPanel("UIHudLeftPanelTeam"):OnDeactivate()
    self.pPanel:SetActive(tbControls.TeamPanelPath, false)
    return
  end
  if self.CurrentPanel == tbWnd.TP_TEAM and self.nPanelState == nPanelType then
    Ui:OpenWindow("UITeam")
    return
  end
  self.CurrentPanel = tbWnd.TP_TEAM
  self.pPanel:SetActive(tbControls.TeamPanelPath, true)
  self.pPanel:SetActive(tbControls.TaskPanelPath, false)
  self.pPanel:SetActive(tbControls.FubenPanelPath, false)
  if self:GetSubPanel("UIHudLeftPanelTeam").OnActivate then
    self:GetSubPanel("UIHudLeftPanelTeam"):OnActivate()
  end
end

function tbWnd:GetSubPanel(szName)
  return self.Panel[szName].pPanel
end

function tbWnd:OnOpen()
  for _, tbPanel in pairs(self.Panel) do
    if tbPanel.pPanel and tbPanel.pPanel.OnOpen then
      tbPanel.pPanel:OnOpen()
    end
  end
  self:SetPanelState(self.panelTaskTeam)
  self:SetButtonState()
  WuLinGaoShou:RequestActivityStatus()
  QinShiHuangLing:RequestActivityStatus()
  Wedding:RequestActivityStatus()
end

function tbWnd:OnOpenEnd()
  local bInFuben, szFubenName = FubenUIController:IsInFubenMap(me)
  if not bInFuben and not WuLinGaoShou.bCanEnterWuLinGaoShou then
    self:SetPanelState(self.panelTaskTeam)
  else
    FubenUIController:OnFubenMapLoaded(me)
  end
  if self.tbOnMapLoadFunc[me.nMapTemplateId] then
    self.tbOnMapLoadFunc[me.nMapTemplateId]()
  end
  self:UpdateChatRoomUi()
end

function tbWnd:OnMapLoaded()
  WuLinGaoShou:RequestActivityStatus()
  QinShiHuangLing:RequestActivityStatus()
  if XiaKeDao:IsXiaKeDaoFuben(me.nMapTemplateId) then
    self.pPanel:SetActive(self.Root, false)
  else
    self.pPanel:SetActive(self.Root, true)
  end
  Wedding:RequestActivityStatus()
  local bInFuben, szFubenName = FubenUIController:IsInFubenMap(me)
  if not bInFuben and not WuLinGaoShou.bCanEnterWuLinGaoShou then
    self:SetPanelState(self.panelTaskTeam)
  else
    FubenUIController:OnFubenMapLoaded(me)
  end
  if self.tbOnMapLoadFunc[me.nMapTemplateId] then
    self.tbOnMapLoadFunc[me.nMapTemplateId]()
  end
  self:UpdateChatRoomUi()
end

function tbWnd:UpdateChatRoomUi()
  local bInLegionChatMap = ChatMgr:InChatMap(me.nMapTemplateId, ChatMgr.tbChatRoomForLegionMap)
  local bInKinAndTongChatMap = ChatMgr:InChatMap(me.nMapTemplateId, ChatMgr.tbChatRoomForKinAndTongMap)
  local bInKinChatMap = ChatMgr:InChatMap(me.nMapTemplateId, ChatMgr.tbChatRoomForKinMap)
  local bInTongChatMap = ChatMgr:InChatMap(me.nMapTemplateId, ChatMgr.tbChatRoomForTongMap)
  local bInTeamChatMap = ChatMgr:InChatMap(me.nMapTemplateId, ChatMgr.tbChatRoomForTeamMap)
  local bHasKinMicPermission = Kin:IsKinVoiceManager()
  local bHasTongMicPermission = Tong:IsTongVoiceManager()
  local bHasLegionMicPermission = Legion:IsLegionVoiceManager()
  local bHasKin = Kin:HasKin()
  local bHasTong = Tong:HasTong()
  local bHasTeam = TeamMgr:HasTeam()
  local bHasLegion = Legion:HasLegion()
  if not bInKinChatMap and not bInTongChatMap and not bInTeamChatMap and not bInKinAndTongChatMap and not bInLegionChatMap then
    self.pPanel:SetActive(self.micOnBt, false)
    self.pPanel:SetActive(self.micOffBt, false)
    self.pPanel:SetActive(self.speakerOnBt, false)
    self.pPanel:SetActive(self.speakerOffBt, false)
    return
  end
  if bInKinAndTongChatMap then
    if bHasTong then
      if not bHasTongMicPermission then
        self.pPanel:SetActive(self.micOnBt, false)
        self.pPanel:SetActive(self.micOffBt, false)
      else
        self.pPanel:SetActive(self.micOnBt, ChatMgr.tbChatRoom.bMicOn)
        self.pPanel:SetActive(self.micOffBt, not ChatMgr.tbChatRoom.bMicOn)
      end
      self.pPanel:SetActive(self.speakerOnBt, ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
      self.pPanel:SetActive(self.speakerOffBt, not ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
    elseif bHasKin then
      if not bHasKinMicPermission then
        self.pPanel:SetActive(self.micOnBt, false)
        self.pPanel:SetActive(self.micOffBt, false)
      else
        self.pPanel:SetActive(self.micOnBt, ChatMgr.tbChatRoom.bMicOn)
        self.pPanel:SetActive(self.micOffBt, not ChatMgr.tbChatRoom.bMicOn)
      end
      self.pPanel:SetActive(self.speakerOnBt, ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
      self.pPanel:SetActive(self.speakerOffBt, not ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
    else
      self.pPanel:SetActive(self.micOnBt, false)
      self.pPanel:SetActive(self.micOffBt, false)
      self.pPanel:SetActive(self.speakerOnBt, false)
      self.pPanel:SetActive(self.speakerOffBt, false)
    end
  elseif bInLegionChatMap then
    if bHasLegion then
      if not bHasLegionMicPermission then
        self.pPanel:SetActive(self.micOnBt, false)
        self.pPanel:SetActive(self.micOffBt, false)
      else
        self.pPanel:SetActive(self.micOnBt, ChatMgr.tbChatRoom.bMicOn)
        self.pPanel:SetActive(self.micOffBt, not ChatMgr.tbChatRoom.bMicOn)
      end
      self.pPanel:SetActive(self.speakerOnBt, ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
      self.pPanel:SetActive(self.speakerOffBt, not ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
    else
      self.pPanel:SetActive(self.micOnBt, false)
      self.pPanel:SetActive(self.micOffBt, false)
      self.pPanel:SetActive(self.speakerOnBt, false)
      self.pPanel:SetActive(self.speakerOffBt, false)
    end
  elseif not ((not bInKinChatMap or bHasKinMicPermission) and (not bInTongChatMap or bHasTongMicPermission)) or bInTeamChatMap and not bHasTeam then
    self.pPanel:SetActive(self.micOnBt, false)
    self.pPanel:SetActive(self.micOffBt, false)
  else
    self.pPanel:SetActive(self.micOnBt, ChatMgr.tbChatRoom.bMicOn)
    self.pPanel:SetActive(self.micOffBt, not ChatMgr.tbChatRoom.bMicOn)
  end
  if bInTongChatMap then
    if not bHasTong then
      self.pPanel:SetActive(self.speakerOnBt, false)
      self.pPanel:SetActive(self.speakerOffBt, false)
    else
      self.pPanel:SetActive(self.speakerOnBt, ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
      self.pPanel:SetActive(self.speakerOffBt, not ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
    end
  end
  if bInKinChatMap then
    if not bHasKin then
      self.pPanel:SetActive(self.speakerOnBt, false)
      self.pPanel:SetActive(self.speakerOffBt, false)
    else
      self.pPanel:SetActive(self.speakerOnBt, ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
      self.pPanel:SetActive(self.speakerOffBt, not ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
    end
  end
  if bInTeamChatMap then
    if not bHasTeam then
      self.pPanel:SetActive(self.speakerOnBt, false)
      self.pPanel:SetActive(self.speakerOffBt, false)
    else
      self.pPanel:SetActive(self.speakerOnBt, ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
      self.pPanel:SetActive(self.speakerOffBt, not ChatMgr.tbChatRoom.bSpeakerOn_PlayerSetting)
    end
  end
end

function tbWnd:SetButtonState()
  if TeamMgr:HasTeam(me) then
    self.pPanel:SetActive(tbControls.TeamButtonGroup, true)
  else
    self.pPanel:SetActive(tbControls.TeamButtonGroup, false)
    return
  end
  if TeamMgr:IsCaptain(me.dwID) then
    self.pPanel:SetActive(tbControls.btnRecruit, true)
  else
    self.pPanel:SetActive(tbControls.btnRecruit, false)
  end
end

function tbWnd:SwitchToTeamUI(szType)
  if szType and "new" == szType and self.nPanelState == self.panelTaskTeam and self.CurrentPanel ~= tbWnd.TP_TEAM then
    for i = 1, #tbWnd.PanelInfo do
      local ref = tbWnd.PanelInfo[i]
      if tbWnd.TP_TEAM == ref.bit then
        if self.pPanel:Toggle_GetChecked(ref.Toggle) == false then
          self.pPanel:Toggle_SetChecked(ref.Toggle, true)
        end
        self.pPanel:SetActive(ref.Obj, true)
        self:GetSubPanel(ref.Name):OnActivate()
      else
        self.pPanel:SetActive(ref.Obj, false)
        self:GetSubPanel(ref.Name):OnDeactivate()
      end
    end
  end
end

  self.pPanel:SetActive(self.micOnBt, true)
  ChatMgr:SwitchMicState(true)

function tbWnd:CloseMic()
  self.pPanel:SetActive(self.micOffBt, true)
  self.pPanel:SetActive(self.micOnBt, false)
  ChatMgr:SwitchMicState(false)
end

function tbWnd:UpdateRedPacketButton(bShow)

    self.pPanel:SetActive(tbControls.btnYuanBaoRedBag, true)

  end
end
