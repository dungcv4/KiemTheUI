function Faction:Init()
  local tbParam = {
    "nId",
    "szName",
    "szIcon",
    "szHudIcon",
    "nSeries",
    "szSeries",
    "nOpenType",
    "szMalePortrait",
    "szFemalePortrait",
    "szRadarChartPath",
    "szMaleAvatar",
    "szFemaleAvatar",
    "szMaleAvatarCenter",
    "szFemaleAvatarCenter",
    "szDesc"
  }
  local szType = "dsssdsdssssssss"
  local tbFactionInfo = Lib:LoadTabFileEx("Setting/Faction/faction.tab", szType, "nId", tbParam)
  if not tbFactionInfo then
    LogErr("error!! Load Faction info faild")
    return
  end
  self.tbFactionInfo = {}
  for i, tbInfo in pairs(tbFactionInfo) do
    if MODULE_GAMECLIENT then
      if tbInfo.szMaleAvatarCenter and tbInfo.szMaleAvatarCenter ~= "" then
        tbInfo.tbMaleAvatarCenter = Lib:SplitStr2Num(tbInfo.szMaleAvatarCenter, ",")
      end
      if tbInfo.szFemaleAvatarCenter and tbInfo.szFemaleAvatarCenter ~= "" then
        tbInfo.tbFemaleAvatarCenter = Lib:SplitStr2Num(tbInfo.szFemaleAvatarCenter, ",")
      end
    end
    tbInfo.szMaleAvatarCenter = nil
    tbInfo.szFemaleAvatarCenter = nil
    self.tbFactionInfo[i] = tbInfo
  end
  self.tbFactionInfo = SetReadOnly(self.tbFactionInfo)
  local tbFactionSectInfo = Lib:EasyLoadTabFile("Setting/Faction/factionsect.tab", {"nFaction", "nSect"})
  if not tbFactionInfo then
    LogErr("error!! Load Faction info faild")
    return
  end
  self.tbFactionSectInfo = tbFactionSectInfo
  self.tbFactionSectInfo = SetReadOnly(self.tbFactionSectInfo)
end

function Faction:IsOpenSex(nFactionId, nSex)
  local tbInfo = self.tbFactionInfo[nFactionId]
  return tbInfo and 0 ~= bit32.band(tbInfo.nOpenType, nSex + 1)
end

function Faction:IsFactionOpen(nFaction)
  return self.tbFactionInfo[nFaction] and 0 ~= self.tbFactionInfo[nFaction].nOpenType
end

function Faction:IsFactionSectOpen(nFaction, nSect)
  if not self:IsFactionOpen(nFaction) then
    return false
  end
  return self.tbFactionSectInfo[nFaction][nSect] ~= nil
end

if not Faction.tbFactionInfo then
  Faction:Init()
end
Faction.MAX_FACTION_COUNT = 1
Faction.CUR_VALID_CAN_SHOW_COUNT = 0
for k, _ in pairs(Faction.tbFactionInfo) do
  if Faction:IsFactionOpen(k) then
    Faction.MAX_FACTION_COUNT = math.max(Faction.MAX_FACTION_COUNT, k)
    Faction.CUR_VALID_CAN_SHOW_COUNT = Faction.CUR_VALID_CAN_SHOW_COUNT + 1
  end
end

if MODULE_GAMECLIENT then
  function Faction:GetOpenDesc(nFaction)
    local tbInfo = self.tbFactionInfo[nFaction]
    
    if not tbInfo then
      return ""
    end
    if Lib:IsEmptyStr(tbInfo.OpenTimeFrame) then
      return ""
    end
    return Calendar:OpenTimeShow(CalcTimeFrameOpenTime(tbInfo.OpenTimeFrame))
  end
