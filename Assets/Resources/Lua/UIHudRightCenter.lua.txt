local tbWnd = Ui:CreateWindow("UIHudRightCenter")

function tbWnd:NotifyEvents()
  return {
    {
      EventNotify.emNOTIFY_MAP_LOADED,
      self.OnMapLoaded,
      self
    },
    {
      EventNotify.emNOTIFY_MAP_LEAVE,
      self.OnLeave,
      self
    },
    {
      EventNotify.emNOTIFY_CHANGE_LEAVE_BUTTON,
      self.OnChangeLeaveButton,
      self
    },
    {
      EventNotify.emNOTIFY_CHANGE_FUBEN_BUTTON,
      self.OnChangeFubenButton,
      self
    },
    {
      EventNotify.emNOTIFY_ONACTIVITY_STATE_CHANGE,
      self.OnActivityStateChange,
      self
    },
    {
      EventNotify.emNOTIFY_PLAYER_LOGIN,
      self.OnPlayerLogin,
      self
    }
  }
end

local tbControls = {
  btnGuide = "imgBG/UIPlayerGuide/Guide",
  UIPlayerGuide = "imgBG/UIPlayerGuide",
  btnLeave = "imgBG/btnLeave",
  btnAFuben = "imgBG/btnAFuben",
  textBtnAFuben = "imgBG/btnAFuben/Text",
  btnStateSwitchCancel = "btnStateSwitchCancel",
  btnStateSwitchCancelText = "btnStateSwitchCancel/Text"
}
tbWnd.tbControls = tbControls
tbWnd.tbOnClick = {
  btnGuide = function(self)
    self:OnClickBtnGuide()
  end,
  btnLeave = function(self)
    local bRet, tbFunc = FubenUIController:IsNeedExitFubenButton()
    if bRet and tbFunc then
      Lib:CallBack(tbFunc)
    else
      FubenUIController:RequireQuit()
    end
  end,
  btnAFuben = function(self)
    local bRet, tbFunc, tbFuncList = FubenUIController:IsNeedFubenBtn()
    if bRet then
      if tbFunc then
        Lib:CallBack(tbFunc)
      elseif tbFuncList then
        Ui:OpenWindowWithCallback("UIRightPopUpMenu3", function(tbSelf, tbWnd)
          local tbPos = self.pPanel:GetTransform(self.btnAFuben).position
          tbWnd:SetPos(tbPos.x - 3, tbPos.y + 0.3)
        end, self, nil, tbFuncList)
      end
    end
  end,
  btnStateSwitchCancel = function(self)
    local bRet, tbFunc = FubenUIController:IsNeedStateCancelBtn()
    if bRet and tbFunc then
      Lib:CallBack(tbFunc)
    end
  end
}

function tbWnd:OnCreate()
  self.PanelPlayerGuide = Ui:CreatePanel("UIPlayerGuide", tbControls.UIPlayerGuide, self)
  self:OnMapLoaded()
end

function tbWnd:OnOpen()
  self.PanelPlayerGuide:OnOpen()
end

function tbWnd:OnClose()
  self.PanelPlayerGuide:OnClose()
end

function tbWnd:OnClickBtnGuide()
  print("touch")
end

function tbWnd:OnChangeLeaveButton()
  self.pPanel:SetActive(tbControls.btnLeave, FubenUIController:IsNeedExitFubenButton())
end

function tbWnd:OnChangeFubenButton()
  self:SetFubenBtnShowStatus(FubenUIController:IsNeedFubenBtn())
  self:SetFubenStateSwitchCancel(FubenUIController:IsNeedStateCancelBtn())
end

function tbWnd:OnMapLoaded()
  self.pPanel:SetActive(tbControls.btnLeave, FubenUIController:IsNeedExitFubenButton())
  self:SetFubenBtnImageAndText(FubenUIController:GetFubenBtnAImagePathAndText())
  self:SetFubenBtnShowStatus(FubenUIController:IsNeedFubenBtn())
  self:SetFubenStateSwitchCancel(FubenUIController:IsNeedStateCancelBtn())
  self.PanelPlayerGuide:OnMapEnter()
end

function tbWnd:OnLeave()
  self.pPanel:SetActive(tbControls.btnLeave, false)
  self.pPanel:SetActive(tbControls.btnAFuben, false)
end

function tbWnd:OnPlayerLogin()
  self.pPanel:SetActive(tbControls.btnLeave, FubenUIController:IsNeedExitFubenButton())
  self:SetFubenBtnImageAndText(FubenUIController:GetFubenBtnAImagePathAndText())
  self:SetFubenBtnShowStatus(FubenUIController:IsNeedFubenBtn())
  self:SetFubenStateSwitchCancel(FubenUIController:IsNeedStateCancelBtn())
end

function tbWnd:OnActivityStateChange()
  self.PanelPlayerGuide:OnActivityStateChange()
end

function tbWnd:SetFubenBtnImageAndText(szImagePath, szText)
  if not szImagePath then
    return
  end
  self.pPanel:Sprite_SetSprite(tbControls.btnAFuben, szImagePath)
  self.pPanel:SetActive(tbControls.textBtnAFuben, false)
  if szText then
    self.pPanel:Label_SetText(tbControls.textBtnAFuben, szText)
    self.pPanel:SetActive(tbControls.textBtnAFuben, true)
  end
end

function tbWnd:SetFubenBtnShowStatus(bShow)
  self.pPanel:SetActive(tbControls.btnAFuben, bShow)
end

function tbWnd:SetFubenBtnText(szText)
  self.pPanel:Label_SetText(tbControls.textBtnAFuben, szText)
end

function tbWnd:SetFubenStateSwitchCancel(bShow, fn, nLeftTime)
  self.pPanel:SetActive(tbControls.btnStateSwitchCancel, bShow)
  if nLeftTime and nLeftTime > 0 then
    self.pPanel:SetActive(tbControls.btnStateSwitchCancelText, true)
    self:SwitchCancelCountDownTimer(nLeftTime)
  else
    self.pPanel:SetActive(tbControls.btnStateSwitchCancelText, false)
  end
end

function tbWnd:SwitchCancelCountDownTimer(nLeftTime)
  if self.nSwitchCancelCountDownTimer then
    Timer:Close(self.nSwitchCancelCountDownTimer)
  end
  self.nCountDown = nLeftTime
  self.nSwitchCancelCountDownTimer = Timer:Register(Env.GAME_FPS, function()
    if Ui:WindowVisible("UIHudRightCenter") then
      return self:UpdateSwitchCancelCountDownText()
    end
  end)
end

function tbWnd:UpdateSwitchCancelCountDownText()
  if not Lib:IsInteger(self.nCountDown) then
    return
  end
  self.nCountDown = self.nCountDown - 1
  self.pPanel:Label_SetText(tbControls.btnStateSwitchCancelText, i18n.EasyGet("\229\137\169\228\189\153\239\188\154{0}", self.nCountDown))
  return self.nCountDown > 0
end
