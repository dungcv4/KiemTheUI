tbWnd.SetHP = {
  [0] = function(self)
    local nLife = 0
    local pNpc = me.GetNpc()
    if pNpc then
      nLife = pNpc.i64CurLife
    end
    self.pPanel:Label_SetText(tbControls.txtHpNum, nLife)
    self.pPanel:Sprite_SetFill(tbControls.imgHP, pNpc.i64CurLife / pNpc.i64MaxLife)
    self.pPanel:SetActive(tbControls.txtHpNum, true)
  end,
  [1] = function(self)
    local nLifePercent = 0
    local pNpc = me.GetNpc()
    if pNpc then
      nLifePercent = pNpc.i64CurLife / pNpc.i64MaxLife
    end
    self.pPanel:Label_SetText(tbControls.txtHpNum, string.format("%d%%", math.floor(nLifePercent * 100)))
    self.pPanel:Sprite_SetFill(tbControls.imgHP, nLifePercent)
    self.pPanel:SetActive(tbControls.txtHpNum, true)
  end,
  [2] = function(self)
    self.pPanel:SetActive(tbControls.txtHpNum, false)
  end
}

function tbWnd:OnVipLevelResp(nPlayerID, nVipLevel)
  local tbData = self.tbInfo or {}
  local tbMember = Kin:GetMemberData(tbData.dwRoleId) or Tong:GetMemberData(tbData.dwRoleId)
  if tbMember then
    tbData.nVipLevel = tbMember.nVipLevel
    tbData.tbChuanGong = tbMember.tbChuanGong
  end
  if nVipLevel and self.nSelectRoleId and nPlayerID and nPlayerID == self.nSelectRoleId then
    self.tbInfo.nVipLevel = nVipLevel
    local tbMemberInfo, nIndex = TeamMgr:GetMember(nPlayerID)
    if tbMemberInfo then
      local pNpc = KNpc.GetById(self.tbInfo.nNpcID)
      if not pNpc or me.GetNpc().GetDistance(pNpc.nId) > Npc.PLAYER_ROLEHEADPOP_DISTANCE then
        Ui:OpenWindow("UIRightPopUpMenu", "chatroleselect", self.tbInfo, self, 700, -140)
        return
      end
    end
    Ui:OpenWindow("UIRightPopUpMenu", "roleselect", self.tbInfo, self, 700, -140)
  end
end

function tbWnd:OnZoneRoleHeadInfoRsp(nPlayerID, tbData)
  Ui:OpenWindow("UIRightPopUpMenu", "zoneroleselect", tbData, self, 700, -140)
end

function tbWnd:DoReqVipLevel(nPlayerID)

function tbWnd:OnCreate()
  self.tbTopPortraitObjPanel = Ui:NewExistPortraitObjPanel(self, tbControls.btnPortrain)
  self.tbPopPortraitObjPanel = Ui:NewExistPortraitObjPanel(self, tbControls.RoleHeadPortraitObj)
  self.pPanel:SetActive(tbControls.imgCADPA, not Client.EnableLoginCADPA)
  self:SetClickEmptyToClose(tbControls.panelPop, true)
  self.pPanel:SetActive(tbControls.panelPop, false)
  self.pPanel:SetActive(tbControls.panelPopNpc, false)
  self.pPanel:SetActive(tbControls.btnOnHook, false)
  self.pPanel:SetActive(tbControls.CountdownNotice, false)
  self.nHpStyle = LocalData:GetRoleNumber("UIHud_TextStyle_HP", 0)
  self.nMaxTimePing = 3
  self.nMaxDelayTime = 1000
  self.nMaxTimeCheckBattery = 300
  self:CheckBattery()
  if self.nBatteryTimer then
    Timer:Close(self.nBatteryTimer)
    self.nBatteryTimer = nil
  end
  self.tbSkilRideCD = self.pPanel:GetObject(tbControls.btnSwitchRide):GetComponent(Ui.UISkillCountDown)
  self.tbSkilRideCD:Reset()
  self.pPanel:SetActive(tbControls.AutoFightFlag, false)
  self:SetAutoPathFlagVisible(false)
  self.bUpdateSkillState = false
  self.pPanel:SetActive(tbControls.BuffTemplate, false)
  self.tbBuffObj = {}
  self.tbBuffComp = {}
  for i = 1, self.MaxBuffCount do
    local itemKey = self.pPanel:CloneObject(tbControls.BuffTemplate, "Buff" .. i)
    local itemGO = self.pPanel:GetObject(itemKey)
    Ui:Button_BindEvent(self.pPanel, itemKey .. "/btnDetail", self.OnClickBuff, self)
    self.tbBuffComp[i] = itemGO:GetComponent(Ui.UIStateCountDown)
    self.pPanel:SetActive(itemKey, false)
    self.tbBuffObj[i] = itemKey
  end
  self.nBatteryTimer = Timer:Register(self.nMaxTimeCheckBattery * Env.GAME_FPS, self.CheckBattery, self)
  if self.nSavePowerTimer then
    Timer:Close(self.nSavePowerTimer)
    self.nSavePowerTimer = nil
  end
  self.nSavePowerTimer = Timer:Register(Env.GAME_FPS, self.CheckSavePower, self)
  self.tbBaitScrollView = Ui:NewScrollView(self, tbControls.btnTemplate, nil, false, false)
  self:StartCheckIdle()
  self.isWaitRedBagUpdate = false
  self:OnHookStop()
  Client.ResourceModule.UnLoadResourceCache()
end

function tbWnd:CheckSavePower()
  Ui:CheckSavePowerMode()
  return true
end

function tbWnd:StartCheckIdle()
  if not Client.nMaxIdleTime then
    return
  end
  self:StopCheckIdle()
  self.nIdleTimer = Timer:Register(Env.GAME_FPS, self.CheckIdle, self)
end

function tbWnd:StopCheckIdle()
  if self.nIdleTimer then
    Timer:Close(self.nIdleTimer)
    self.nIdleTimer = nil
  end
end

function tbWnd:IsIdleCheckOpening()
  if not OnOffController:Check(OnOffController.KEY.OnlineIdle) then
    return false
  end
  return true
end

function tbWnd:CheckIdle()
  if Ui.GameEnv.EnableAutoTest then
    return
  end
  if not self:IsIdleCheckOpening() then
    return true
  end
  Ui:CheckIdle()
  return true
end

  local bCustomSkill = FightSkill:IsCustomSkillStatus(me)
  self:UpdateRideButton(bFactionSkill, bCustomSkill)
  self:UpdateRideButtonMark()

  self:OnMpChange()
  self:UpdateAllSkillState()
  self:UpdateTime()
  self:UpdateAFKState()
  self.tbRideHorseInfo = self.tbRideHorseInfo or {}
  self:OnUpdateRideHorseCD()
  self:UpdateRideHorseUsePoint()
  self:SendPing()
  if self.nPingTimer then
    Timer:Close(self.nPingTimer)
    self.nPingTimer = nil
  end
  self.nPingTimer = Timer:Register(self.nMaxTimePing * Env.GAME_FPS, self.SendPing, self)
  self.tbInfo = tbInfo
  self:UpdateUpHorseBtnState()
  self:RequestPortrait()
  self.tbTopPortraitObjPanel:SetLevel(me.nTotalLevel)
  self:ShowOrHideFightModePanel(false, true)
  self:OnPKModeChange()

    self:SetAutoPathFlagVisible(true)
  end
  self:UpdateFightPower(Stronger.nFightPower)
  self:UpdatePlayerExpInfo()
  self:UpdateRedBagInfo()

  RedPointMgr:DynamicRegisteRedPoint(self.pPanel, tbControls.btnStrong, {
    RedPointMgr.EVENT.Stronger1,
    RedPointMgr.EVENT.Stronger2,
    RedPointMgr.EVENT.Stronger3
  })

    RedPointMgr.EVENT.PotentialPointLeft,
    RedPointMgr.EVENT.FashionPrettyStageAward,
    RedPointMgr.EVENT.NewPortrait,
    RedPointMgr.EVENT.NewPortraitFrame,
    RedPointMgr.EVENT.NewSpeechBalloon,
    RedPointMgr.EVENT.NewChatTitlePrefix,
    RedPointMgr.EVENT.ShenBingBagEntrance
  }

  RedPointMgr:ChangeRedPointPositionLeft(self.pPanel, tbControls.headRedPoint)

  self:SetIconsShow()

function tbWnd:OnClose()
  Ui:LeaveSavePowerMode()
  if self.nSavePowerTimer then
    Timer:Close(self.nSavePowerTimer)
    self.nSavePowerTimer = nil
  end
  self:StopCheckIdle()
  Ui:CloseWindow("UIJoyStick")
  self.nLastPortrait = nil
  if self.nPingTimer then
    Timer:Close(self.nPingTimer)
    self.nPingTimer = nil
  end
  if self.nTimeTimerId then
    Timer:Close(self.nTimeTimerId)
    self.nTimeTimerId = nil
  end
  if self.nNotifyMsgTimerID then
    Timer:Close(self.nNotifyMsgTimerID)
    self.nNotifyMsgTimerID = nil
  end
  if self.nBatteryTimer then
    Timer:Close(self.nBatteryTimer)
    self.nBatteryTimer = nil
  end
  if self.nReqPortraitTimer then
    Timer:Close(self.nReqPortraitTimer)
    self.nReqPortraitTimer = nil
  end
  if self.nRedBagTimer then
    Timer:Close(self.nRedBagTimer)
    self.nRedBagTimer = nil
  end
  if self.nRedBagCloseIconTimer then
    Timer:Close(self.nRedBagCloseIconTimer)
    self.nRedBagCloseIconTimer = nil
  end
  self:SetCountdownNotice()
  RedPointMgr:DynamicUnRegist(self.pPanel, tbControls.btnStrong)
  RedPointMgr:DynamicUnRegist(self.pPanel, tbControls.btnRole)
  RedPointMgr:DynamicUnRegist(self.pPanel, tbControls.btnPortrain)
end

  if me.IsMaskPlayerAttr() then
    return
  end
  me.SwitchRide(state)
  if Map:IsForbidRide(me.nMapTemplateId) and 1 == state then
    me.CenterMsg(i18n.EasyGet("\229\189\147\229\137\141\229\156\176\229\155\190\228\184\141\229\133\129\232\174\184\228\184\138\233\169\172"))
  end
  if 1 == state and 1 == HanHaiMiCheng:GetKeyStatus() then
    me.CenterMsg(i18n.EasyGet("\229\189\147\229\137\141\231\138\182\230\128\129\228\184\141\229\133\129\232\174\184\228\184\138\233\169\172"))
  end
  self:OnUpdateRideHorseCD()
  self:UpdateRideHorseUsePoint()
end

end

function tbWnd:UpdateRideButton(bFactionSkill, bCustomSkill)
  if nil == bFactionSkill then
    bFactionSkill = FightSkill:IsFactionSkillStatus(me)
  end
  if nil == bCustomSkill then
    bCustomSkill = FightSkill:IsCustomSkillStatus(me)
  end
  if not bFactionSkill and not bCustomSkill then
    self.pPanel:SetActive(tbControls.btnSwitchRide, false)
    self.pPanel:SetActive(tbControls.btnSwitchRideDown, false)
    if 1 == me.nRideState then
      SetRideState(0)
    end
    return
  end
  self.pPanel:SetActive(tbControls.btnSwitchRide, 0 == me.nRideState)
  self.pPanel:SetActive(tbControls.btnSwitchRideDown, 0 ~= me.nRideState)
end

function tbWnd:UpdateRideButtonMark()

  self.pPanel:SetActive(tbControls.btnSwitchRide .. "/KeyboardMark", 1 == nSetting)
  self.pPanel:SetActive(tbControls.btnSwitchRideDown .. "/KeyboardMark", 1 == nSetting)
  self.pPanel:SetActive("Left/btnSit/KeyboardMark", 1 == nSetting)
end

function tbWnd:SetRideButtonNotActive()

  self.pPanel:SetActive(tbControls.btnSwitchRideDown, false)
end

function tbWnd:ShowRoleHeadPop(tbInfo)
  self.tbPopPortraitObjPanel:SetRoleIcon(tbInfo.nPortrait)
  self.tbPopPortraitObjPanel:SetLevel(tbInfo.nTotalLevel or tbInfo.nLevel)
  self.tbPopPortraitObjPanel:SetFactionIcon(tbInfo.nFaction)
  self.nPlayerID = tbInfo.dwRoleId
  self.nPlayerNpcId = tbInfo.nNpcID
  self.tbInfo = tbInfo
  self.pPanel:SetActive(tbControls.panelPop, self:IsShowRoleHeadPop())
  local tbRequest = {
    szClass = "UIHud",
    szCallBackFunc = "OnPlayerPortraitFrameDataArrival",
    tbRequestMember = {}
  }
  tbRequest.tbRequestMember = {
    self.nPlayerID
  }
  RemoteServer.PortraitFrameCmd("ApplyGroupPlayersPFWithCB", tbRequest)
end

function tbWnd:OnPlayerPortraitFrameDataArrival(tbData)
  if not tbData[self.nPlayerID] then
    return false
  end
  local nPortraitFrameID = tbData[self.nPlayerID]
  self.tbPopPortraitObjPanel:SetPortraitFrame(nPortraitFrameID)
end

function tbWnd:GetSignal()
  local dwDelayTime = GetPingDelay()
  local fFillDelay = (self.nMaxDelayTime - dwDelayTime) / self.nMaxDelayTime
  if fFillDelay <= 1.0E-4 then
    fFillDelay = 1.0E-4
  end
  if 1.0E-4 == fFillDelay then
    self.pPanel:Label_SetText(tbControls.txtPing, string.format("<color=#b60000>%s</color>", i18n.EasyGet("{0}\230\175\171\231\167\146", dwDelayTime)))
    return 0
  elseif fFillDelay <= 0.25 then
    self.pPanel:Label_SetText(tbControls.txtPing, string.format("<color=#b60000>%s</color>", i18n.EasyGet("{0}\230\175\171\231\167\146", dwDelayTime)))
    return 0.25
  elseif fFillDelay <= 0.5 then
    self.pPanel:Label_SetText(tbControls.txtPing, string.format("<color=#ffff00>%s</color>", i18n.EasyGet("{0}\230\175\171\231\167\146", dwDelayTime)))
    return 0.5
  elseif fFillDelay <= 0.75 then
    self.pPanel:Label_SetText(tbControls.txtPing, string.format("<color=#ffff00>%s</color>", i18n.EasyGet("{0}\230\175\171\231\167\146", dwDelayTime)))
    return 0.75
  else
    self.pPanel:Label_SetText(tbControls.txtPing, string.format("<color=#28ff28>%s</color>", i18n.EasyGet("{0}\230\175\171\231\167\146", dwDelayTime)))
    return 1
  end
end

function tbWnd:UpdateRideHorseUsePoint()
  local pNpc = me.GetNpc()
  if not pNpc then
    return
  end
  local tbInfo = self.tbRideHorseInfo
  if not tbInfo then
    return
  end
  local bIsUsePoint, nUsePoint, nPerUsePoint = pNpc.GetRideHorseUsePoint()

    self.tbSkilRideCD:SetUsePoint(bIsUsePoint, nUsePoint, nPerUsePoint)

function tbWnd:UpdateRideHorseCD(bNotify)
  local pNpc = me.GetNpc()
  if not pNpc then
    return
  end
  local tbInfo = self.tbRideHorseInfo
  if not tbInfo then
    return
  end
  local nNextFrame, nCurCDTime = pNpc.GetRideHorseNextCastTime()
  if nNextFrame <= 0 then
    self.nRideHorseCDCacheTime = nil
    if self.tbSkilRideCD then
      self.tbSkilRideCD:Reset()
    end
    return
  end
  local nLogicFrame = GetFrame()
  local nFrame = nNextFrame - nLogicFrame
  if nFrame < 0 then
    nFrame = 0
  end
  local fTotalCDTime = nCurCDTime / Env.GAME_FPS
  local fTime = nFrame / Env.GAME_FPS
  local fTotalTime = fTotalCDTime > 0 and fTotalCDTime or fTime
  if tbInfo.bUseTotalTime then
    fTotalTime = tbInfo.fTotalTime
  end
  local nCdfTime = fTime
  local nCdfTotalTime = fTotalTime
  local nStartFrame = self.nRideHorseCDCacheTime
  if fTotalTime < 1 or not nStartFrame then
    self.nRideHorseCDCacheTime = nil
    nCdfTime = 0
    nCdfTotalTime = 0
  else
    nCdfTotalTime = fTotalCDTime
  end
  local bIsUsePoint, nUsePoint, nPerUsePoint = pNpc.GetRideHorseUsePoint()
  if self.tbSkilRideCD then
    self.tbSkilRideCD:Run(bIsUsePoint, nUsePoint, nPerUsePoint, nCdfTime, nCdfTotalTime)
  end
  tbInfo.fTotalTime = fTotalTime
end

function tbWnd:OnPKModeChange()
  local tbButton = {
    [1] = tbControls.btnSwitchPeace,
    [2] = tbControls.btnSwitchButcher,
    [3] = tbControls.btnSwitchKin,
    [4] = tbControls.btnSwitchTong
  }
  local nShowButton
  if me.nPkMode == Npc.PK_MODE.emPK_MODE_PEACE then
    nShowButton = 1
  elseif me.nPkMode == Npc.PK_MODE.emPK_MODE_KILLER then
    nShowButton = 2
  elseif me.nPkMode == Npc.PK_MODE.emPK_MODE_KIN then
    if 0 == me.dwTongId then
      nShowButton = 3
    else
      nShowButton = 4
    end
  elseif me.nPkMode == Npc.PK_MODE.emPK_MODE_CUSTOM then
    self.pPanel:SetActive(tbControls.panelSwitchFightMode, false)
    nShowButton = 1
    if me.nMapTemplateId == GlobalHangUp.KongQueHeMapTemplateID then
      if 0 == me.dwTongId then
        nShowButton = 3
      else
        nShowButton = 4
      end
    end
  end
  for i = 1, #tbButton do
    self.pPanel:SetActive(tbButton[i], nShowButton == i)
  end
  self:SwitchKinOrTongButtonState()
end

  if self.bUpdateSkillState then
    self.bUpdateSkillState = false
    local pNpc = me.GetNpc()
    
    local function fnSort(tbA, tbB)
      local nABuffSkillId = pNpc.GetSkillState(tbA.nSkillId).nSkillId
      local nBBuffSkillId = pNpc.GetSkillState(tbB.nSkillId).nSkillId
      local nASkillLevel = pNpc.GetSkillState(tbA.nSkillId).nSkillLevel
      local nBSkillLevel = pNpc.GetSkillState(tbB.nSkillId).nSkillLevel
      local tbAStateEffect = FightSkill:GetStateEffectBySkill(nABuffSkillId, nASkillLevel)
      local tbBStateEffect = FightSkill:GetStateEffectBySkill(nBBuffSkillId, nBSkillLevel)
      assert(tbAStateEffect and tbBStateEffect)
      if tbAStateEffect.Sort ~= tbBStateEffect.Sort then
        return tbAStateEffect.Sort < tbBStateEffect.Sort
      end
      if tbAStateEffect.StateEffectId ~= tbBStateEffect.StateEffectId then
        return tbAStateEffect.StateEffectId < tbBStateEffect.StateEffectId
      end
    end
    
    table.sort(self.tbSkillStateInfo, fnSort)
    self:RefreshBuff()
    local pNpc = me.GetNpc()
    for nIndex, tbInfo in pairs(self.tbSkillStateInfo) do
      local tbState = pNpc.GetSkillState(tbInfo.nSkillId)
      if tbState and self.tbBuffObj[nIndex] ~= nil then
        local nFrame = tbState.nEndFrame - GetFrame()
        local tbStateEffect = FightSkill:GetStateEffectBySkill(tbState.nSkillId, tbState.nSkillLevel)
        assert(tbStateEffect)
        self.pPanel:SetActive(self.tbBuffObj[nIndex], true)
        self.pPanel:Sprite_SetSprite(self.tbBuffObj[nIndex], tbStateEffect.Icon)
        local fTime = 0
        local fTotalTime = 1
        if nFrame >= 0 then
          fTime = nFrame / Env.GAME_FPS
          fTotalTime = tbState.nTotalFrame / Env.GAME_FPS
        end
        local tbSkillInfo = FightSkill:GetSkillSetting(tbState.nSkillId, tbState.nSkillLevel)
        if not tbSkillInfo.IsAura and -1 ~= tbState.nEndFrame then
          self.tbBuffComp[nIndex]:Run(fTime, fTotalTime)
        else
          self.tbBuffComp[nIndex]:Stop(false)
        end
        local nShowLv
        for _, tbMagic in pairs(tbState.tbAttrib) do
          if tbMagic.szName == "superposemagic" and tbMagic.tbValue[2] > 0 then
            nShowLv = tbMagic.tbValue[2]
            break
          end
        end
        if not nShowLv and tbState.nStateMultRecord > 0 then
          nShowLv = tbState.nStateMultRecord
        end
        local nItemKey = self.tbBuffObj[nIndex] .. "/txtBuffLv"
        if nShowLv then
          self.pPanel:Label_SetText(nItemKey, tostring(nShowLv))
          self.pPanel:SetActive(nItemKey, true)
        else
          self.pPanel:SetActive(nItemKey, false)
        end
      end
    end
  end
end

function tbWnd:OnClickBuff()
  self:RefreshBuff()
  if not self.tbSkillStateInfo or not next(self.tbSkillStateInfo) then
    return
  end
  Ui:OpenWindow("UIBuffTip", self.tbSkillStateInfo)
end

function tbWnd:ClearAllSkillState(bNotClearInfo)
  for i = 1, self.MaxBuffCount do
    self.pPanel:SetActive(self.tbBuffObj[i], false)
  end
  if not bNotClearInfo then
    self.tbSkillStateInfo = {}
  end
end

function tbWnd:RefreshBuff()
  local pNpc = me.GetNpc()
  if not pNpc then
    return
  end
  local tbRemoveIds = {}
  for _, tbInfo in pairs(self.tbSkillStateInfo) do
    local tbState = pNpc.GetSkillState(tbInfo.nSkillId)
    if not tbState or tbState.nEndFrame <= 0 and -1 ~= tbState.nEndFrame then
      tbRemoveIds[tbInfo.nSkillId] = true
    end
  end
  for k, _ in pairs(tbRemoveIds) do
    self:RemoveSkillState(k)
  end
end

function tbWnd:SortSkillState()
  self:RefreshBuff()
  if not self.tbSkillStateInfo or not next(self.tbSkillStateInfo) then
    return
  end
  self.bUpdateSkillState = true
end

function tbWnd:GetSkillStateIndex(nCurSkillId)
  for nIndex, tbInfo in pairs(self.tbSkillStateInfo) do
    if tbInfo.nSkillId == nCurSkillId then
      return nIndex, true
    end
  end
  local nTotalCount = #self.tbSkillStateInfo
  return nTotalCount + 1, false
end

function tbWnd:AddSkillState(tbState)
  if not tbState then
    return
  end
  if not FightSkill:IsShowSkillStateIcon(tbState.nSkillId, tbState.nSkillLevel) then
    return
  end
  local nIndex = self:GetSkillStateIndex(tbState.nSkillId)
  self.tbSkillStateInfo[nIndex] = {
    nSkillId = tbState.nSkillId
  }
end

function tbWnd:UpdateAllSkillState()
  self:ClearAllSkillState()
  local pNpc = me.GetNpc()
  if not pNpc then
    return
  end
  local tbAllSkillState = pNpc.GetAllSkillState() or {}
  for _, tbState in pairs(tbAllSkillState) do
    self:AddSkillState(tbState)
    self:SortSkillState()
  end
end

function tbWnd:OnAddSkillState(nSkillId)
  local pNpc = me.GetNpc()
  local tbState = pNpc.GetSkillState(nSkillId)
  self:AddSkillState(tbState)
  self:SortSkillState()
  self:RefreshBuff()
end

function tbWnd:OnRemoveSkillState(nSkillId)
  self:RemoveSkillState(nSkillId)
end

function tbWnd:RemoveSkillState(nSkillId)
  local nIndex, bExist = self:GetSkillStateIndex(nSkillId)
  if not bExist then
    return
  end
  if nSkillId == Kin.Def.TongModeStateEffectId then
    self.bTongModeState = false
  end
  table.remove(self.tbSkillStateInfo, nIndex)
  self:ResetAddSkillState()
end

function tbWnd:ResetAddSkillState()
  self:ClearAllSkillState(true)
  local pNpc = me.GetNpc()
  for _, tbInfo in pairs(self.tbSkillStateInfo) do
    local tbState = pNpc.GetSkillState(tbInfo.nSkillId)
    self:AddSkillState(tbState)
    self:SortSkillState()
  end
end

function tbWnd:UpdateCountdownNotice()
  local now = GetTime()
  local nEndTime = self.nCountdownNoticeEndTime or 0
  if now > nEndTime then
    self.pPanel:SetActive(tbControls.CountdownNotice, false)
    self.nCountdownNoticeTimerID = nil
    return false
  else
    local nCountdown = nEndTime - now
    local szText = string.format(self.szCountdownNoticeText, Lib:TimeDesc(nCountdown))
    self.pPanel:SetActive(tbControls.CountdownNotice, true)
    self.pPanel:Label_SetText(tbControls.CountdownNotice, szText)
    return true
  end
end

function tbWnd:SetCountdownNotice(nEndTime, szText)
  if nEndTime then
    self.nCountdownNoticeEndTime = nEndTime
    self.szCountdownNoticeText = szText
    if not self.nCountdownNoticeTimerID then
      self:UpdateCountdownNotice()
      self.nCountdownNoticeTimerID = Timer:Register(Env.GAME_FPS * 1, function()
        return self:UpdateCountdownNotice()
      end)
    end
  else
    if self.nCountdownNoticeTimerID then
      Timer:Close(self.nCountdownNoticeTimerID)
      self.nCountdownNoticeTimerID = nil
    end
    self.nCountdownNoticeEndTime = 0
    self.szCountdownNoticeText = ""
    self:UpdateCountdownNotice()
  end
end

function tbWnd:UpdateBait()
  self.tbBaitScrollView:Reset()
  local pInstance = ActivityMgr:GetInstanceByClassName(ActivityMgr.Def.ClassNameList.SummerFishing)
  if not pInstance then
    return
  end
  local tbSetting = pInstance:GetDialogSetting()
  for nIndex, tbInfo in ipairs(tbSetting) do
    self.tbBaitScrollView:SetElement(nIndex, function(tbElement)
      tbElement.pPanel:Label_SetText("Btn/Text", tbInfo.szName)
      tbElement.pPanel:Label_SetText("BtnSelected/Text", tbInfo.szName)
      if tbInfo.bGrey then
        tbElement.pPanel:SetActive("BtnSelected", false)
        tbElement.pPanel:SetActive("Btn", true)
        tbElement.pPanel:Button_SetEnable("Btn", false)
      elseif tbInfo.bSelected then
        tbElement.pPanel:SetActive("BtnSelected", true)
        tbElement.pPanel:SetActive("Btn", false)
      else
        tbElement.pPanel:SetActive("BtnSelected", false)
        tbElement.pPanel:SetActive("Btn", true)
        tbElement.pPanel:Button_SetEnable("Btn", true)
      end
      Ui:Button_BindEvent(tbElement.pPanel, "Btn", self.OnClickBaitBtn, self, nIndex)
      Ui:Button_BindEvent(tbElement.pPanel, "BtnSelected", self.OnClickBaitBtn, self, nIndex)
    end)
  end
end
